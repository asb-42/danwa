Pydantic schemas for profile management.

Defines typed, validated models for LLM profiles, agent personas, prompt variants, and active debate configurations.

 30class LLMProfile(BaseModel):
 31    """Configuration for a specific LLM endpoint."""
 32
 33    id: str = Field(default="", pattern=r"^([a-z0-9][a-z0-9.-]*)?$")
 34    name: str
 35    profile_type: Literal["text", "tts", "stt"] = "text"
 36    provider: LLMProvider
 37    model: str  # e.g. "anthropic/claude-3.5-sonnet"
 38    api_base: str | None = None  # For OpenRouter / local
 39    api_key_env: str = "OPENROUTER_API_KEY"  # Environment variable name
 40    api_key: str | None = None  # BYOK: Direct API key (takes precedence over env var)
 41    account_id_env: str | None = None  # Environment variable name for account ID (e.g. Cloudflare)
 42    max_tokens: int = 4096
 43    context_window: int | None = None  # Max total tokens (input + output) the model supports
 44    temperature: float = 0.7
 45    timeout: int = 600
 46
 47    # Cost tracking (USD per 1k tokens)
 48    cost_per_1k_input: float | None = None
 49    cost_per_1k_output: float | None = None
 50
 51    # --- A2A Protocol (Phase 8) ---
 52    protocol: Literal["litellm", "a2a"] = "litellm"
 53    a2a_endpoint: str | None = None  # URL for A2A agent (e.g. "http://agent.example.com")
 54    a2a_timeout: int = 120  # Timeout for A2A calls in seconds
 55    fallback_llm_profile_id: str | None = None  # Fallback profile for A2A failures
 56
 57    # --- Service LLM (Sprint 16) ---
 58    service_eligible: bool = True  # Whether this profile can be used for system/background tasks
 59    min_recommended_context: int = 1024  # Min context window recommended for service use
 60
 61    @field_validator("temperature")
 62    @classmethod
 63    def validate_temperature(cls, v: float) -> float:
 64        if not 0 <= v <= 2:
 65            raise ValueError("Temperature must be between 0 and 2")
 66        return v
 67
 68    @field_validator("max_tokens")
 69    @classmethod
 70    def validate_max_tokens(cls, v: int) -> int:
 71        if v < 1:
 72            raise ValueError("max_tokens must be at least 1")
 73        return v

 82    system_prompt: str
 83

 87

 91
 92    # LLM profile binding
 93    llm_profile_id: str | None = None
 94
 95    # Metadata
 96    description: str | None = None
 97    tags: list[str] = []

100    @classmethod
101    def validate_threshold(cls, v: float) -> float:
102        if not 0 <= v <= 1:
103            raise ValueError("consensus_threshold must be between 0 and 1")
104        return v
