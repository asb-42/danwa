backend.api.routers.workflow_exec

Workflow Execution — API router for running workflows and interjections.

Endpoints:

  • POST /{workflow_id}/start — starts workflow execution
  • GET /{session_id}/state — returns current execution state
  • POST /{session_id}/pause — pauses execution
  • POST /{session_id}/resume — resumes execution
  • POST /{session_id}/cancel — cancels execution
  • GET /{session_id}/stream — SSE endpoint for real-time events
  • POST /{session_id}/interject — submit a user interjection during execution

140@router.post("/{workflow_id}/start", response_model=StartWorkflowResponse)
141async def start_workflow(
142    workflow_id: str,
143    body: StartWorkflowRequest,
144    background_tasks: BackgroundTasks,
145    project_store: ProjectStore = Depends(get_project_store),
146) -> StartWorkflowResponse:
147    """Start executing a workflow definition.
148
149    Loads the WorkflowDefinition from the repository, compiles it into
150    a LangGraph StateGraph, builds the initial state, and launches
151    the execution as a background task.
152    """
153    repo = _get_repo()
154    snapshot_store = _get_snapshot_store()
155
156    # Load workflow definition
157    workflow = repo.get_workflow_definition(workflow_id)
158    if workflow is None:
159        raise HTTPException(status_code=404, detail=f"Workflow '{workflow_id}' not found")
160
161    # Compile to LangGraph
162    compiler = CompilerService(repo)
163    compiled = compiler.compile_to_langgraph(workflow)
164
165    if not compiled.is_valid:
166        raise HTTPException(
167            status_code=422,
168            detail={
169                "message": "Workflow compilation failed",
170                "errors": compiled.errors,
171                "warnings": compiled.warnings,
172            },
173        )
174
175    # Generate session ID
176    session_id = f"wf-{uuid.uuid4().hex[:12]}"
177
178    # Resolve RAG context
179    rag_context = ""
180    if body.document_ids or body.rag_auto_retrieve:
181        try:
182            from backend.services.debate_workflow import resolve_rag_context
183
184            rag_context, _ = resolve_rag_context(
185                project_id=body.project_id,
186                case_text=body.context,
187                document_ids=body.document_ids,
188                rag_auto_retrieve=body.rag_auto_retrieve,
189            )
190            if rag_context:
191                logger.info(
192                    "RAG context resolved for workflow %s (%d chars)",
193                    workflow_id,
194                    len(rag_context),
195                )
196        except Exception:
197            logger.warning(
198                "Failed to resolve RAG context for workflow %s",
199                workflow_id,
200                exc_info=True,
201            )
202
203    # Build initial state
204    initial_state: dict[str, Any] = {
205        "workflow_id": workflow_id,
206        "session_id": session_id,
207        "project_id": body.project_id,
208        "context": body.context,
209        "language": body.language,
210        "rag_context": rag_context,
211        "node_sequence": compiled.node_sequence,
212        "node_configs": {
213            agent.node_id: {
214                "blueprint_id": agent.blueprint_id,
215                "blueprint_name": agent.blueprint_name,
216                "llm_profile_id": agent.llm_profile_id,
217                "llm_model": agent.llm_model,
218                "role_definition_id": agent.role_definition_id,
219                "role": agent.role,
220                "prompt_template_id": agent.prompt_template_id,
221            }
222            for agent in compiled.resolved_agents
223        },
224        "edge_map": {},
225        "termination_conditions": [],
226        "current_node_id": "",
227        "current_round": 1,
228        "max_rounds": body.max_rounds,
229        "threshold": body.threshold,
230        "node_outputs": [],
231        "messages": [],
232        "current_draft": "",
233        "interjection_queue": [],
234        "consumed_interjections": [],
235        "final_consensus": 0.0,
236        "output": "",
237        "status": "running",
238        "is_paused": False,
239        "pause_event": get_pause_event(session_id),
240    }
241
242    set_session_status(session_id, "running")
243
244    # Launch as background task
245    background_tasks.add_task(
246        run_workflow_background,
247        session_id=session_id,
248        workflow_id=workflow_id,
249        project_id=body.project_id,
250        initial_state=initial_state,
251        compiled_workflow=compiled,
252        snapshot_store=snapshot_store,
253    )
254
255    logger.info("Started workflow %s as session %s", workflow_id, session_id)
256    return StartWorkflowResponse(session_id=session_id, status="running")

259@router.get("/{session_id}/state", response_model=SessionStateResponse)
260async def get_session_state(session_id: str) -> SessionStateResponse:
261    """Get the current execution state for a workflow session.
262
263    Returns the latest state snapshot from SQLite.
264    """
265    snapshot_store = _get_snapshot_store()
266    status = get_session_status(session_id)
267
268    if status == "unknown":
269        # Try to load from snapshot store
270        snapshot = snapshot_store.get_latest(session_id)
271        if snapshot is None:
272            raise HTTPException(status_code=404, detail=f"Session '{session_id}' not found")
273        state = snapshot.get("state", {})
274        return SessionStateResponse(
275            session_id=session_id,
276            status=state.get("status", "unknown"),
277            workflow_id=snapshot.get("workflow_id"),
278            current_node_id=snapshot.get("node_id"),
279            current_round=snapshot.get("round_number", 0),
280            node_outputs=state.get("node_outputs", []),
281            output=state.get("output"),
282            final_consensus=state.get("final_consensus"),
283        )
284
285    # Load latest snapshot for running/paused sessions
286    snapshot = snapshot_store.get_latest(session_id)
287    state = snapshot.get("state", {}) if snapshot else {}
288
289    return SessionStateResponse(
290        session_id=session_id,
291        status=status,
292        workflow_id=snapshot.get("workflow_id") if snapshot else None,
293        current_node_id=snapshot.get("node_id") if snapshot else None,
294        current_round=snapshot.get("round_number", 0) if snapshot else 0,
295        node_outputs=state.get("node_outputs", []),
296        output=state.get("output"),
297        final_consensus=state.get("final_consensus"),
298    )

302async def pause_workflow(session_id: str) -> StatusResponse:
303    """Pause a running workflow.
304

307    guard_mutable(session_id)

310        raise HTTPException(

312            detail=f"Cannot pause session in status '{status}'",
313        )
314
315    pause_session(session_id)
316    logger.info("Paused session %s", session_id)
317    return StatusResponse(session_id=session_id, status="paused")

339@router.post("/{session_id}/cancel", response_model=StatusResponse)
340async def cancel_workflow(session_id: str) -> StatusResponse:
341    """Cancel a running or paused workflow.
342
343    Sets a cancellation flag that the workflow checks between nodes.
344    Idempotent: if already completed/failed/cancelled, returns current status.
345    """
346    guard_mutable(session_id)
347    status = get_session_status(session_id)
348    if status in ("completed", "failed", "cancelled"):
349        return StatusResponse(session_id=session_id, status=status)
350
351    cancel_session(session_id)
352
353    # Also resume if paused, so the workflow can exit
354    resume_session(session_id)
355
356    logger.info("Cancelled session %s", session_id)
357    return StatusResponse(session_id=session_id, status="cancelled")

360@router.get("/{session_id}/stream")
361async def stream_workflow_events(session_id: str) -> EventSourceResponse:
362    """SSE endpoint for real-time workflow execution events.
363
364    Subscribes to the event bus and yields events as they are published
365    by workflow nodes.
366    """
367    status = get_session_status(session_id)
368
369    async def event_generator():
370        # Send initial status
371        yield {
372            "event": "status",
373            "data": json.dumps({"session_id": session_id, "status": status}),
374        }
375
376        # If already terminal, just return
377        if status in ("completed", "failed", "cancelled"):
378            return
379
380        # Subscribe to event bus
381        queue = subscribe(session_id)
382        try:
383            while True:
384                try:
385                    event_type, payload = await asyncio.wait_for(queue.get(), timeout=300.0)
386                    yield {"event": event_type, "data": payload}
387
388                    # Stop on terminal events
389                    if event_type in ("workflow.complete", "node.error"):
390                        break
391                except TimeoutError:
392                    # Send keepalive
393                    yield {"event": "ping", "data": "{}"}
394        finally:
395            unsubscribe(session_id, queue)
396
397    return EventSourceResponse(event_generator())

400@router.post("/{session_id}/interject", response_model=InterjectResponse)
401async def submit_interjection(
402    session_id: str,
403    body: InterjectRequest,
404) -> InterjectResponse:
405    """Submit a user interjection for a running workflow session.
406
407    The interjection is queued and will be consumed by the next
408    interjection node in the workflow graph.  Emits an SSE event
409    for frontend visualization.
410    """
411    guard_mutable(session_id)
412    interjection_id = await interjection_service.submit(
413        session_id=session_id,
414        content=body.content,
415        source=body.source,
416        metadata=body.metadata,
417    )
418
419    # Emit SSE event for frontend visualization
420    await publish_async(
421        session_id,
422        "interjection.received",
423        {
424            "type": "interjection.received",
425            "interjection_id": interjection_id,
426            "content": body.content,
427            "source": body.source,
428        },
429    )
430
431    # --- Audit log ---
432    try:
433        get_audit_logger().log_interjection(
434            session_id=session_id,
435            workflow_id="",
436            workflow_version=1,
437            actor=body.source,
438            content=body.content,
439            metadata={"interjection_id": interjection_id, **(body.metadata or {})},
440        )
441    except Exception:
442        logger.debug("Audit logging failed for interjection", exc_info=True)
443
444    logger.info("Interjection %s queued for session %s", interjection_id, session_id)
445    return InterjectResponse(
446        interjection_id=interjection_id,
447        status="queued",
448    )

@router.get('/{session_id}/stream')
async def stream_workflow_events(session_id: str) -> sse_starlette.sse.EventSourceResponse: 
361@router.get("/{session_id}/stream")
362async def stream_workflow_events(session_id: str) -> EventSourceResponse:
363    """SSE endpoint for real-time workflow execution events.
364
365    Subscribes to the event bus and yields events as they are published
366    by workflow nodes.
367    """
368    status = get_session_status(session_id)
369
370    async def event_generator():
371        # Send initial status
372        yield {
373            "event": "status",
374            "data": json.dumps({"session_id": session_id, "status": status}),
375        }
376
377        # If already terminal, just return
378        if status in ("completed", "failed", "cancelled"):
379            return
380
381        # Subscribe to event bus
382        queue = subscribe(session_id)
383        try:
384            while True:
385                try:
386                    event_type, payload = await asyncio.wait_for(queue.get(), timeout=300.0)
387                    yield {"event": event_type, "data": payload}
388
389                    # Stop on terminal events
390                    if event_type in ("workflow.complete", "node.error"):
391                        break
392                except TimeoutError:
393                    # Send keepalive
394                    yield {"event": "ping", "data": "{}"}
395        finally:
396            unsubscribe(session_id, queue)
397
398    return EventSourceResponse(event_generator())
