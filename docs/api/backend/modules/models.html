backend.modules.models

Pydantic models for the Danwa Module System v2.

v2 changes:

  • 1 module = 1 profile (not a bundle)
  • profile_file + profile_format replace files[] for single-profile modules
  • Legacy files[] still supported during migration
  • New typed models for RoleType, ToneProfile

145class ModuleManifest(BaseModel):
146    """Manifest for a Danwa module.
147
148    v2 (single-profile): uses `profile_file` + `profile_format`
149    v1 (legacy bundle): uses `files[]`
150    """
151
152    schema_version: str = "2.0.0"
153    module_id: str = Field(..., pattern=r"^[a-z][a-z0-9.-]*$")
154    name: dict[str, str] = Field(default_factory=dict)
155    description: dict[str, str] = Field(default_factory=dict)
156    version: str = "1.0.0"
157    type: ModuleType
158    category: ModuleCategory
159    author: dict[str, str] = Field(default_factory=dict)
160    license: str = "CC-BY-4.0"
161    dependencies: dict[str, str] = Field(default_factory=dict)
162    tags: list[str] = Field(default_factory=list)
163    language: str = "en"
164    checksum: str = ""
165
166    # v2: single profile file
167    profile_file: str | None = None  # e.g. "profile.yaml"
168    profile_format: Literal["yaml", "json"] | None = None
169
170    # v1: legacy bundle files (migration compat)
171    files: list[ModuleFile] = Field(default_factory=list)
172
173    created_at: datetime | None = None
174    updated_at: datetime | None = None
175
176    @field_validator("module_id")
177    @classmethod
178    def validate_module_id(cls, v: str) -> str:
179        v = v.replace("_", "-")
180        if not v.startswith("danwa-") and "-" not in v:
181            raise ValueError(f"module_id must contain at least one hyphen (non-danwa module), got '{v}'")
182        return v
