 59    def _resolve_module_dir(self, module_id: str) -> Path | None:
 60        """Find a module directory by ID, searching root and one level of subdirectories.
 61
 62        First tries direct path match (modules/<module_id>/manifest.json),
 63        then falls back to reading all manifest.json files to check the
 64        ``module_id`` field (since directory names may differ from the
 65        module_id stored in the manifest).
 66        """
 67        direct = self.modules_dir / module_id
 68        if (direct / "manifest.json").exists():
 69            return direct
 70        for subdir in self.modules_dir.iterdir():
 71            if subdir.is_dir() and not subdir.name.startswith("."):
 72                candidate = subdir / module_id
 73                if (candidate / "manifest.json").exists():
 74                    return candidate
 75                # Also check the subdir itself as a module directory
 76                manifest_path = subdir / "manifest.json"
 77                if manifest_path.exists():
 78                    try:
 79                        data = json.loads(manifest_path.read_text(encoding="utf-8"))
 80                        if data.get("module_id") == module_id:
 81                            return subdir
 82                    except (json.JSONDecodeError, OSError):
 83                        continue
 84                # Check one more level (subdir/<child>/) for matching module_id
 85                for child in sorted(subdir.iterdir()):
 86                    if not child.is_dir() or child.name.startswith("."):
 87                        continue
 88                    child_manifest = child / "manifest.json"
 89                    if child_manifest.exists():
 90                        try:
 91                            data = json.loads(child_manifest.read_text(encoding="utf-8"))
 92                            if data.get("module_id") == module_id:
 93                                return child
 94                        except (json.JSONDecodeError, OSError):
 95                            continue
 96        return None

 98    def discover_local(self) -> list[ModuleInfo]:
 99        modules: list[ModuleInfo] = []
100        if not self.modules_dir.exists():
101            return modules
102
103        # Search root and one level of subdirectories (category dirs)
104        search_dirs = []
105        for entry in sorted(self.modules_dir.iterdir()):
106            if not entry.is_dir() or entry.name.startswith("."):
107                continue
108            if (entry / "manifest.json").exists():
109                search_dirs.append(entry)
110            else:
111                for sub in sorted(entry.iterdir()):
112                    if sub.is_dir() and not sub.name.startswith("."):
113                        search_dirs.append(sub)
114
115        for module_dir in search_dirs:
116            if not (module_dir / "manifest.json").exists():
117                continue
118            try:
119                info = self._dir_to_info(module_dir)
120                if info:
121                    modules.append(info)
122            except Exception:
123                logger.exception("Failed to read module %s", module_dir.name)
124
125        return modules

204    def fetch_repo_index(
205        self,
206        repo_url: str = DANWA_MODULES_INDEX_URL,
207        force_refresh: bool = False,
208    ) -> list[dict[str, Any]]:
209        """Fetch the ``index.json`` from the danwa-modules repository.
210
211        Returns a list of module entries, each with ``module_id``,
212        ``version``, ``type``, ``download_url``, ``checksum_sha256``,
213        and optional ``translation_stats``.
214
215        Results are cached for 24 hours (configurable via
216        ``_registry_cache_ttl``).
217        """
218        import time
219
220        now = time.time()
221        if not force_refresh and self._registry_cache and (now - self._registry_cache_time) < self._registry_cache_ttl:
222            return self._registry_cache.get("modules", [])
223
224        try:
225            req = urllib.request.Request(
226                repo_url,
227                headers={"User-Agent": "Danwa/2.1.0 (ModuleService)"},
228            )
229            with urllib.request.urlopen(req, timeout=15) as resp:
230                data = json.loads(resp.read().decode("utf-8"))
231            self._registry_cache = data
232            self._registry_cache_time = now
233            return data.get("modules", [])
234        except Exception as exc:
235            logger.warning("Module repo index not reachable (%s): %s", repo_url, exc)
236            return []

242    def install_from_repo(
243        self,
244        module_id: str,
245        version: str | None = None,
246    ) -> InstallationReport:
247        """Install a module directly from the danwa-modules GitHub release.
248
249        Args:
250            module_id: The module ID to install.
251            version: Specific version to install (defaults to latest).
252
253        Returns:
254            InstallationReport with the result.
255
256        Raises:
257            FileNotFoundError: If the module or version is not found in the index.
258        """
259        index = self.fetch_repo_index(force_refresh=True)
260        candidates = [m for m in index if m["module_id"] == module_id]
261        if not candidates:
262            raise FileNotFoundError(f"Module '{module_id}' not found in danwa-modules repository")
263
264        target = candidates[0]
265        if version:
266            matches = [m for m in candidates if m.get("version") == version]
267            if not matches:
268                raise FileNotFoundError(f"Module '{module_id}' version {version} not found in danwa-modules repository")
269            target = matches[0]
270
271        module_version = target.get("version", "0.0.0")
272        download_url = target.get(
273            "download_url",
274            self.get_download_url(module_id, module_version),
275        )
276
277        checksum = target.get("checksum_sha256", "")
278
279        # Pre-flight dependency check
280        installed_local = self.discover_local()
281        installed_map = {m.module_id: m.version for m in installed_local}
282
283        errors = self.dependency_resolver.resolve(
284            module_id,
285            target.get("dependencies", {}),
286            installed_map,
287        )
288        if errors:
289            report = InstallationReport(
290                status="error",
291                module_id=module_id,
292                version=module_version,
293                errors=errors,
294            )
295            return report
296
297        report = self.installer.install_from_url(download_url)
298        if report.status == "ok" and checksum and not report.checksum:
299            report.checksum = checksum
300        return report

302    def check_updates(
303        self,
304        repo_url: str = DANWA_MODULES_INDEX_URL,
305    ) -> list[dict[str, Any]]:
306        """Compare installed module versions against the danwa-modules repo index.
307
308        Uses semver comparison — any remote version greater than the
309        installed version is reported as an available update.
310
311        Returns:
312            List of ``{module_id, current_version, available_version,
313            download_url, checksum_sha256}`` dicts.
314        """
315        remote_modules = self.fetch_repo_index(repo_url)
316        installed = {m.module_id: m for m in self.discover_local()}
317
318        updates = []
319        for remote_mod in remote_modules:
320            mod_id = remote_mod.get("module_id", "")
321            if mod_id not in installed:
322                continue
323
324            local_version = installed[mod_id].version
325            remote_version = remote_mod.get("version", "0.0.0")
326
327            try:
328                local_ver = Version(local_version)
329                remote_ver = Version(remote_version)
330                if remote_ver > local_ver:
331                    updates.append(
332                        {
333                            "module_id": mod_id,
334                            "current_version": local_version,
335                            "available_version": remote_version,
336                            "download_url": remote_mod.get(
337                                "download_url",
338                                self.get_download_url(mod_id, remote_version),
339                            ),
340                            "checksum_sha256": remote_mod.get("checksum_sha256", ""),
341                            "name": remote_mod.get("name", {}),
342                        }
343                    )
344            except Exception:
345                # If version parsing fails, fall back to string comparison
346                if remote_version != local_version:
347                    updates.append(
348                        {
349                            "module_id": mod_id,
350                            "current_version": local_version,
351                            "available_version": remote_version,
352                            "download_url": remote_mod.get("download_url", ""),
353                            "checksum_sha256": remote_mod.get("checksum_sha256", ""),
354                            "name": remote_mod.get("name", {}),
355                        }
356                    )
357
358        updates.sort(key=lambda u: u["module_id"])
359        return updates

384    def _force_uninstall(self, module_id: str) -> UninstallationReport:
385        target_dir = self._resolve_module_dir(module_id)
386        if not target_dir:
387            return UninstallationReport(
388                status="error",
389                module_id=module_id,
390                errors=[f"Module directory not found: {module_id}"],
391            )
392        files_removed = 0
393        if target_dir.exists():
394            for f in target_dir.rglob("*"):
395                if f.is_file():
396                    files_removed += 1
397            shutil.rmtree(target_dir)
398
399        for bak in self.modules_dir.glob(f"{module_id}.bak.*"):
400            if bak.is_dir():
401                shutil.rmtree(bak)
402
403        db_entries_removed = 0
404        try:
405            conn = self.installer._get_db()
406            cursor = conn.cursor()
407            cursor.execute("DELETE FROM module_translation_cache WHERE module_id = ?", (module_id,))
408            db_entries_removed += cursor.rowcount
409            cursor.execute("DELETE FROM module_registry WHERE id = ?", (module_id,))
410            db_entries_removed += cursor.rowcount
411            conn.commit()
412            conn.close()
413        except sqlite3.Error as e:
414            logger.error("Database error during force uninstall of %s: %s", module_id, e)
415            return UninstallationReport(
416                status="error",
417                module_id=module_id,
418                errors=[f"Database error: {e}"],
419            )
420
421        logger.info("Force-uninstalled module %s (%d files, %d DB entries)", module_id, files_removed, db_entries_removed)
422
423        return UninstallationReport(
424            status="ok",
425            module_id=module_id,
426            files_removed=files_removed,
427            db_entries_removed=db_entries_removed,
428        )

437    def get_profile(self, module_id: str) -> dict[str, Any] | None:
438        module_dir = self._resolve_module_dir(module_id)
439        if not module_dir:
440            return None
441        manifest_path = module_dir / "manifest.json"
442        if not manifest_path.exists():
443            return None
444
445        manifest = json.loads(manifest_path.read_text(encoding="utf-8"))
446        profile_file = manifest.get("profile_file")
447        profile_format = self._derive_profile_format(profile_file, manifest.get("profile_format"))
448
449        if not profile_file:
450            return None
451
452        profile_path = module_dir / profile_file
453        if not profile_path.exists():
454            return None
455
456        content = profile_path.read_text(encoding="utf-8")
457        if profile_format == "yaml":
458            return yaml.safe_load(content)
459        elif profile_format == "json":
460            return json.loads(content)
461        elif profile_format == "markdown":
462            return {"content": content}
463        return None

465    def update_profile(self, module_id: str, profile_data: dict[str, Any]) -> bool:
466        module_dir = self._resolve_module_dir(module_id)
467        if not module_dir:
468            return False
469        manifest_path = module_dir / "manifest.json"
470        if not manifest_path.exists():
471            return False
472
473        manifest = json.loads(manifest_path.read_text(encoding="utf-8"))
474        profile_file = manifest.get("profile_file")
475        profile_format = self._derive_profile_format(profile_file, manifest.get("profile_format"))
476
477        if not profile_file:
478            return False
479
480        # Fields that belong in manifest.json rather than the profile file
481        manifest_fields = {"name", "description", "role", "tags", "language", "version"}
482        manifest_dirty = False
483        for key in manifest_fields:
484            if key in profile_data:
485                val = profile_data[key]
486                # Wrap name/description into language dict if given as plain string
487                if key in ("name", "description") and isinstance(val, str):
488                    val = {"en": val}
489                manifest[key] = val
490                manifest_dirty = True
491
492        if manifest_dirty:
493            manifest_path.write_text(
494                json.dumps(manifest, indent=2, ensure_ascii=False) + "\n",
495                encoding="utf-8",
496            )
497
498        safe_data = {k: v for k, v in profile_data.items() if k not in {"id"} | manifest_fields}
499
500        profile_path = module_dir / profile_file
501        if profile_format == "yaml":
502            profile_path.write_text(yaml.dump(safe_data, default_flow_style=False, sort_keys=False, allow_unicode=True), encoding="utf-8")
503        elif profile_format == "json":
504            profile_path.write_text(json.dumps(safe_data, indent=2, ensure_ascii=False), encoding="utf-8")
505        elif profile_format == "markdown":
506            profile_path.write_text(safe_data.get("content", ""), encoding="utf-8")
507        else:
508            return False
509
510        self._update_manifest_checksum(module_dir, manifest)
511        return True

513    def duplicate_module(self, module_id: str, new_id: str, new_name: str | None = None) -> dict[str, Any] | None:
514        src_dir = self._resolve_module_dir(module_id)
515        dst_dir = self.modules_dir / new_id
516
517        if not src_dir or dst_dir.exists():
518            return None
519
520        shutil.copytree(src_dir, dst_dir)
521
522        manifest_path = dst_dir / "manifest.json"
523        manifest = json.loads(manifest_path.read_text(encoding="utf-8"))
524        manifest["module_id"] = new_id
525        if "created_at" not in manifest:
526            manifest["created_at"] = datetime.now(UTC).isoformat()
527        manifest["updated_at"] = datetime.now(UTC).isoformat()
528        if new_name:
529            manifest["name"]["en"] = new_name
530            manifest["name"]["de"] = new_name
531
532        profile_file = manifest.get("profile_file")
533        if profile_file:
534            profile_path = dst_dir / profile_file
535            if profile_path.exists():
536                profile_format = manifest.get("profile_format")
537                new_profile_id = uuid.uuid4().hex[:8]
538                if profile_format == "yaml":
539                    data = yaml.safe_load(profile_path.read_text(encoding="utf-8"))
540                    data["id"] = new_profile_id
541                    if new_name:
542                        data["name"] = new_name
543                    profile_path.write_text(yaml.dump(data, default_flow_style=False, sort_keys=False, allow_unicode=True), encoding="utf-8")
544                elif profile_format == "json":
545                    data = json.loads(profile_path.read_text(encoding="utf-8"))
546                    data["id"] = new_profile_id
547                    if new_name:
548                        data["name"] = new_name
549                    profile_path.write_text(json.dumps(data, indent=2, ensure_ascii=False), encoding="utf-8")
550                elif profile_format == "markdown":
551                    pass
552
553        manifest_path.write_text(json.dumps(manifest, indent=2, ensure_ascii=False), encoding="utf-8")
554
555        self.installer.install_from_directory(dst_dir)
556        return self._dir_to_info(dst_dir)

558    def translate(
559        self,
560        module_id: str,
561        target_lang: str,
562        force: bool = False,
563        llm_profile_id: str | None = None,
564        skip_back_translation: bool = False,
565        auto_approve: bool = False,
566        quality_threshold: float = 0.7,
567    ) -> TranslationResult:
568        try:
569            conn = sqlite3.connect(str(self.db_path))
570            conn.row_factory = sqlite3.Row
571            cursor = conn.cursor()
572
573            cursor.execute(
574                "SELECT file_path, source_hash FROM module_translation_cache WHERE module_id = ? AND (source_language = 'en' OR language = 'en')",
575                (module_id,),
576            )
577            source_files = cursor.fetchall()
578
579            if not source_files:
580                conn.close()
581                return TranslationResult(
582                    module_id=module_id,
583                    target_language=target_lang,
584                    status="error",
585                    errors=["No source (EN) content found for module"],
586                )
587
588            translated = 0
589            skipped = 0
590            quality_scores: dict[str, float] = {}
591
592            for row in source_files:
593                fpath = row["file_path"]
594                source_hash = row["source_hash"]
595
596                if not force:
597                    cursor.execute(
598                        "SELECT quality_score FROM module_translation_cache WHERE module_id = ? AND file_path = ? AND language = ?",
599                        (module_id, fpath, target_lang),
600                    )
601                    existing = cursor.fetchone()
602                    if existing:
603                        skipped += 1
604                        quality_scores[fpath] = existing["quality_score"]
605                        continue
606
607                cursor.execute(
608                    """
609                    INSERT OR REPLACE INTO module_translation_cache
610                        (id, module_id, file_path, language,
611                         translated_content, source_hash, quality_score,
612                         generated_at, generated_by, approved)
613                    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
614                    """,
615                    (
616                        f"{module_id}:{fpath}:{target_lang}",
617                        module_id,
618                        fpath,
619                        target_lang,
620                        None,
621                        source_hash,
622                        0.0,
623                        datetime.now(UTC).isoformat(),
624                        "system",
625                        0,
626                    ),
627                )
628                translated += 1
629                quality_scores[fpath] = 0.0
630
631            conn.commit()
632            conn.close()
633

492
493        if manifest_dirty:
494            manifest_path.write_text(
495                json.dumps(manifest, indent=2, ensure_ascii=False) + "\n",
496                encoding="utf-8",
497            )
498
499        safe_data = {k: v for k, v in profile_data.items() if k not in {"id"} | manifest_fields}
500
501        profile_path = module_dir / profile_file
502        if profile_format == "yaml":
503            profile_path.write_text(yaml.dump(safe_data, default_flow_style=False, sort_keys=False, allow_unicode=True), encoding="utf-8")
504        elif profile_format == "json":
