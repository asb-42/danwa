Produces a structured report including:

  • Debate title, case description, metadata
  • Per-round agent outputs (full transcript)
  • Consensus progression
  • Final consensus and result
  • Audit trail table (optional, if entries exist)

 32def _display_agent_role(role: str) -> str:
 33    """Format an agent role for display.
 34
 35    MVP debates pass pre-formatted names like ``"Strategist (deepseek-v4-flash)"``
 36    while legacy debates pass raw role names like ``"critic"``.  Capitalize
 37    only when the role starts with a lowercase letter.
 38    """
 39    if not role:
 40        return "Unbekannt"
 41    return role if role[0].isupper() else role.capitalize()

 44def _build_mvp_rounds_from_snapshot(debate_data: dict[str, Any]) -> list[dict[str, Any]]:
 45    """Build round data from state snapshot for MVP debates.
 46
 47    MVP debates don't populate ``debate_data["rounds"]`` — the agent
 48    outputs live exclusively in the state snapshot's ``node_outputs``.
 49    This helper loads the snapshot, resolves agent names and LLM
 50    profiles from ``node_configs``, and returns a list of round dicts
 51    compatible with the existing renderer expectations.
 52
 53    Returns:
 54        A list of round dicts with keys ``round``, ``consensus``,
 55        ``agent_outputs``.
 56    """
 57    session_id = debate_data.get("session_id", "")
 58    if not session_id:
 59        return []
 60
 61    try:
 62        snap_store = StateSnapshotStore()
 63        snapshot = snap_store.get_latest(session_id)
 64    except Exception:
 65        logger.warning("Could not load state snapshot for session %s", session_id)
 66        return []
 67
 68    if not snapshot:
 69        return []
 70
 71    state = snapshot.get("state", {})
 72    node_outputs: list[dict] = state.get("node_outputs", [])
 73    node_configs_raw: dict[str, Any] = state.get("node_configs", {})
 74    llm_assignments: dict[str, str] = debate_data.get("llm_assignments", {})
 75
 76    if not node_outputs:
 77        return []
 78
 79    # Parse node_configs — values may be stored as Python repr strings
 80    config_by_node: dict[str, dict] = {}
 81    for nid, cfg in node_configs_raw.items():
 82        if isinstance(cfg, str):
 83            try:
 84                cfg = ast.literal_eval(cfg)
 85            except (ValueError, SyntaxError):
 86                cfg = {}
 87        config_by_node[nid] = cfg if isinstance(cfg, dict) else {}
 88
 89    # Build agent outputs with proper names and metadata
 90    agent_outputs: list[dict[str, Any]] = []
 91    for no in node_outputs:
 92        role = no.get("role", "")
 93        node_id = no.get("node_id", "")
 94        config = config_by_node.get(node_id, {})
 95
 96        llm_model = config.get("llm_model", "")
 97        llm_profile_id = config.get("llm_profile_id", "") or llm_assignments.get(role, "")
 98        role_type_name = config.get("role_type_name", "")
 99
100        # Build a human-readable agent name: "Strategist (deepseek-v4-flash)"
101        agent_name = role_type_name or role.replace("_", " ").title()
102        if llm_model:
103            agent_name = f"{agent_name} ({llm_model})"
104
105        agent_outputs.append(
106            {
107                "role": agent_name,
108                "content": no.get("content", ""),
109                "tokens_used": no.get("tokens_used", 0),
110                "duration_ms": no.get("duration_ms", 0),
111                "llm_profile_id": llm_profile_id,
112                "round": no.get("round"),
113            }
114        )
115
116    current_round = state.get("current_round", debate_data.get("current_round", 1))
117
118    return [
119        {
120            "round": current_round,
121            "consensus": state.get("final_consensus", debate_data.get("result", {}).get("consensus", 0.0)),
122            "agent_outputs": agent_outputs,
123        }
124    ]

130    def __init__(self, db_path: Path | str | None = None) -> None:
131        self._db_path = Path(db_path) if db_path else None
132        self._audit = AuditLogger(self._db_path)
133        _REPORTS_DIR.mkdir(parents=True, exist_ok=True)

135    async def generate(
136        self,
137        session_id: str,
138        fmt: str = "docx",
139        debate_data: dict[str, Any] | None = None,
140    ) -> Path:
141        """Generate a report for *session_id* in the given *fmt*.
142
143        Args:
144            session_id: The workflow session ID (same as debate_id).
145            fmt: Output format — ``"docx"``, ``"pdf"``, or ``"odf"``.
146            debate_data: Optional debate dict from DebateStore.  When
147                provided the report includes the full debate transcript.
148
149        Returns:
150            Path to the generated report file.
151        """
152        if fmt not in ("docx", "pdf", "odf"):
153            raise ValueError(f"Unsupported format: {fmt!r}")
154
155        ts = datetime.now().strftime("%Y%m%d_%H%M%S")
156        filename = f"report_{session_id[:8]}_{ts}.{fmt}"
157        path = _REPORTS_DIR / filename
158
159        # Load audit data (supplementary)
160        audit_entries = self._audit.get_audit_log_for_replay(session_id)
161
162        # Build transcript data from debate store
163        transcript = self._build_transcript(debate_data)
164
165        if fmt == "docx":
166            await asyncio.to_thread(self._build_docx, session_id, transcript, audit_entries, path)
167        elif fmt == "pdf":
168            await asyncio.to_thread(self._build_pdf, session_id, transcript, audit_entries, path)
169        elif fmt == "odf":
170            await asyncio.to_thread(self._build_odf, session_id, transcript, audit_entries, path)
171
172        logger.info("Report generated: %s", path)
173        return path

179    @staticmethod
180    def _build_transcript(debate_data: dict[str, Any] | None) -> dict[str, Any]:
181        """Extract structured transcript data from a debate dict.
182
183        Returns a normalized dict with keys: title, case_text, language,
184        model, max_rounds, threshold, status, current_round,
185        final_consensus, rounds (list of round dicts).
186        """
187        if not debate_data:
188            return {
189                "title": "",
190                "case_text": "",
191                "language": "de",
192                "model": "",
193                "max_rounds": 0,
194                "threshold": 0.0,
195                "status": "unknown",
196                "current_round": 0,
197                "final_consensus": 0.0,
198                "rounds": [],
199                "output": "",
200            }
201
202        req = debate_data.get("request", {})
203        result = debate_data.get("result", {})
204        rounds = debate_data.get("rounds", [])
205
206        # rounds may come from either the debate store or the result dict
207        if not rounds and result:
208            rounds = result.get("rounds", [])
209
210        # MVP debates: build rounds from state snapshot node_outputs
211        is_mvp = debate_data.get("is_mvp", False)
212        if is_mvp and not rounds:
213            rounds = _build_mvp_rounds_from_snapshot(debate_data)
214
215        return {
216            "title": debate_data.get("title", ""),
217            "case_text": req.get("case", {}).get("text", "") if isinstance(req.get("case"), dict) else str(req.get("case", "")),
218            "language": req.get("language", "de"),
219            "model": req.get("llm_profile_id", ""),
220            "max_rounds": req.get("max_rounds", 0),
221            "threshold": req.get("consensus_threshold", 0.0),
222            "status": str(debate_data.get("status", "unknown")),
223            "current_round": debate_data.get("current_round", result.get("current_round", 0)),
224            "final_consensus": result.get("final_consensus", debate_data.get("final_consensus", 0.0)),
225            "rounds": rounds,
226            "output": result.get("output", ""),
227            "is_mvp": is_mvp,
228        }

234    @staticmethod
235    def _extract_transactional_protocol(
236        audit_entries: list[dict[str, Any]],
237    ) -> list[dict[str, Any]]:
238        """Extract CriticItem → BuildResponse → PragmatistEvaluation data from audit entries.
239
240        Parses ``builder_iteration`` and ``pragmatist_evaluation`` workflow
241        events from the audit log and merges them into a unified protocol
242        table for the report.
243        """
244        import json as _json
245
246        iterations: dict[int, dict] = {}  # draft_version → builder data
247        evaluations: list[dict] = []
248
249        for entry in audit_entries:
250            event_type = entry.get("event_type", "")
251            raw_output = entry.get("output_content", "")
252
253            if event_type == "builder_iteration":
254                try:
255                    if isinstance(raw_output, str) and raw_output:
256                        data = _json.loads(raw_output)
257                    elif isinstance(raw_output, dict):
258                        data = raw_output
259                    else:
260                        data = {}
261                except (ValueError, TypeError):
262                    data = {}
263                dv = entry.get("draft_version", data.get("draft_version", 0))
264                iterations[dv] = {
265                    "draft_version": dv,
266                    "constructivity_score": entry.get("constructivity_score", data.get("constructivity_score")),
267                    "build_response_count": data.get("build_response_count", 0),
268                }
269
270            elif event_type == "pragmatist_evaluation":
271                try:
272                    if isinstance(raw_output, str) and raw_output:
273                        data = _json.loads(raw_output)
274                    elif isinstance(raw_output, dict):
275                        data = raw_output
276                    else:
277                        data = {}
278                except (ValueError, TypeError):
279                    data = {}
280                verdicts = data.get("verdicts", [])
281                for v in verdicts:
282                    evaluations.append(v)
283
284        # Build merged protocol entries
285        protocol: list[dict[str, Any]] = []
286
287        for ev in evaluations:
288            resp_to = ev.get("response_to", "")
289            verdict = ev.get("verdict", "pending")
290            feasibility = ev.get("feasibility")
291
292            # Find the iteration this belongs to
293            draft_ver = 0
294            for dv, it_data in sorted(iterations.items()):
295                draft_ver = dv
296
297            protocol.append(
298                {
299                    "critic_item_id": resp_to,
300                    "build_response_summary": f"Verdict: {verdict}",
301                    "verdict": verdict,
302                    "feasibility": feasibility,
303                    "draft_version": draft_ver,
304                }
305            )
306
307        # If no pragmatist evaluations, fall back to builder iterations
308        if not protocol and iterations:
309            for dv, it_data in sorted(iterations.items()):
310                protocol.append(
311                    {
312                        "critic_item_id": f"Iteration {dv}",
313                        "build_response_summary": f"{it_data.get('build_response_count', 0)} Antworten",
314                        "verdict": "pending",
315                        "feasibility": it_data.get("constructivity_score"),
316                        "draft_version": dv,
317                    }
318                )
319
320        return protocol

326    def _build_docx(
327        self,
328        session_id: str,
329        transcript: dict[str, Any],
330        audit_entries: list[dict[str, Any]],
331        path: Path,
332    ) -> None:
333        doc = Document()
334        doc.styles["Normal"].font.name = "Calibri"
335
336        # --- Title ---
337        title = transcript["title"] or f"Debatte {session_id[:8]}"
338        doc.add_heading(title, level=0)
339
340        # --- Metadata ---
341        doc.add_heading("Zusammenfassung", level=1)
342        table_meta = doc.add_table(rows=0, cols=2, style="Table Grid")
343        meta_rows = [
344            ("Session-ID", session_id),
345            ("Titel", transcript["title"]),
346            ("Sprache", transcript["language"]),
347            ("Modell", transcript["model"]),
348            ("Max. Runden", str(transcript["max_rounds"])),
349            ("Konsens-Schwelle", f"{transcript['threshold']:.0%}"),
350            ("Status", transcript["status"]),
351            ("Runden absolviert", str(transcript["current_round"])),
352            ("Finaler Konsens", f"{transcript['final_consensus']:.1%}"),
353            ("Generiert", datetime.now().strftime("%Y-%m-%d %H:%M:%S")),
354        ]
355        for label, value in meta_rows:
356            row = table_meta.add_row()
357            row.cells[0].text = label
358            row.cells[1].text = value
359
360        # --- Case description ---
361        case_text = transcript["case_text"]
362        if case_text:
363            doc.add_heading("Fallbeschreibung", level=1)
364            doc.add_paragraph(case_text)
365
366        # --- Debate transcript (per round) ---
367        rounds = transcript["rounds"]
368        if rounds:
369            doc.add_heading("Debatte-Transkript", level=1)
370            for rd in rounds:
371                round_num = rd.get("round", "?")
372                consensus = rd.get("consensus", 0.0)
373                doc.add_heading(f"Runde {round_num} — Konsens: {consensus:.1%}", level=2)
374
375                agent_outputs = rd.get("agent_outputs", [])
376                for ao in agent_outputs:
377                    role = ao.get("role", "unbekannt")
378                    content = ao.get("content", "")
379                    tokens = ao.get("tokens_used", 0)
380                    duration_ms = ao.get("duration_ms", 0)
381                    llm_pid = ao.get("llm_profile_id", "")
382                    heading = _display_agent_role(role)
383                    if llm_pid:
384                        heading += f" — {llm_pid}"
385                    doc.add_heading(heading, level=3)
386                    if content:
387                        for paragraph in content.split("\n\n"):
388                            paragraph = paragraph.strip()
389                            if paragraph:
390                                doc.add_paragraph(paragraph)
391                    meta_parts = []
392                    if tokens:
393                        meta_parts.append(f"Tokens: {tokens}")
394                    if duration_ms:
395                        meta_parts.append(f"Latenz: {duration_ms}ms")
396                    if llm_pid:
397                        meta_parts.append(f"LLM-Profil: {llm_pid}")
398                    if meta_parts:
399                        doc.add_paragraph(" | ".join(meta_parts)).italic = True
400
401        # --- Final output ---
402        output = transcript["output"]
403        if output:
404            doc.add_heading("Ergebnis", level=1)
405            for paragraph in output.split("\n\n"):
406                paragraph = paragraph.strip()
407                if paragraph:
408                    doc.add_paragraph(paragraph)
409
410        # --- Transactional Drafting: Konstruktionsprotokoll ---
411        txn_entries = self._extract_transactional_protocol(audit_entries)
412        if txn_entries:
413            doc.add_heading("Konstruktionsprotokoll", level=1)
414
415            # Table: CriticItem → BuildResponse → PragmatistEvaluation
416            table = doc.add_table(rows=1, cols=3, style="Table Grid")
417            headers = ["CriticItem", "BuildResponse (Optionen)", "Pragmatist-Evaluation"]
418            for i, h in enumerate(headers):
419                table.rows[0].cells[i].text = h
420
421            for entry in txn_entries:
422                row = table.add_row()
423                row.cells[0].text = entry.get("critic_item_id", "")
424                row.cells[1].text = entry.get("build_response_summary", "")
425                verdict = entry.get("verdict", "pending")
426                feasibility = entry.get("feasibility")
427                cell_text = verdict.upper()
428                if feasibility is not None:
429                    cell_text += f"\nMachbarkeit: {feasibility * 100:.1f}%"
430                row.cells[2].text = cell_text
431
432        # --- Audit trail (supplementary) ---
433        if audit_entries:
434            doc.add_heading("Audit-Trail", level=1)
435            table = doc.add_table(rows=1, cols=7, style="Table Grid")
436            headers = ["Zeitstempel", "Ereignis", "Knoten", "Aktor", "LLM-Profil", "Latenz (ms)", "Tokens"]
437            for i, h in enumerate(headers):
438                table.rows[0].cells[i].text = h
439            for entry in audit_entries:
440                row = table.add_row()
441                row.cells[0].text = str(entry.get("timestamp", ""))
442                row.cells[1].text = str(entry.get("event_type", ""))
443                row.cells[2].text = str(entry.get("node_id", ""))
444                row.cells[3].text = str(entry.get("actor", ""))
445                row.cells[4].text = str(entry.get("llm_profile_id", ""))
446                row.cells[5].text = str(entry.get("latency_ms", 0))
447                row.cells[6].text = str(entry.get("prompt_tokens", 0) + entry.get("completion_tokens", 0))
448
449        doc.save(str(path))

469    def _build_odf(
470        self,
471        session_id: str,
472        transcript: dict[str, Any],
473        audit_entries: list[dict[str, Any]],
474        path: Path,
475    ) -> None:
476        html_content = self._render_html(session_id, transcript, audit_entries)
477        try:
478            from odf.opendocument import OpenDocumentText
479            from odf.text import H, P
480
481            doc = OpenDocumentText()
482
483            title = transcript["title"] or f"Debatte {session_id[:8]}"
484            doc.text.addElement(H(text=title, outlinelevel=1))
485
486            # Metadata
487            doc.text.addElement(H(text="Zusammenfassung", outlinelevel=2))
488            meta_lines = [
489                f"Session-ID: {session_id}",
490                f"Titel: {transcript['title']}",
491                f"Sprache: {transcript['language']}",
492                f"Modell: {transcript['model']}",
493                f"Max. Runden: {transcript['max_rounds']}",
494                f"Konsens-Schwelle: {transcript['threshold']:.0%}",
495                f"Status: {transcript['status']}",
496                f"Runden absolviert: {transcript['current_round']}",
497                f"Finaler Konsens: {transcript['final_consensus']:.1%}",
498            ]
499            for line in meta_lines:
500                doc.text.addElement(P(text=line))
501
502            # Case description
503            case_text = transcript["case_text"]
504            if case_text:
505                doc.text.addElement(H(text="Fallbeschreibung", outlinelevel=2))
506                for para in case_text.split("\n\n"):
507                    para = para.strip()
508                    if para:
509                        doc.text.addElement(P(text=para))
510
511            # Transcript
512            rounds = transcript["rounds"]
513            if rounds:
514                doc.text.addElement(H(text="Debatte-Transkript", outlinelevel=2))
515                for rd in rounds:
516                    round_num = rd.get("round", "?")
517                    consensus = rd.get("consensus", 0.0)
518                    doc.text.addElement(H(text=f"Runde {round_num} — Konsens: {consensus:.1%}", outlinelevel=3))
519                    for ao in rd.get("agent_outputs", []):
520                        role = ao.get("role", "unbekannt")
521                        content = ao.get("content", "")
522                        llm_pid = ao.get("llm_profile_id", "")
523                        role_label = _display_agent_role(role)
524                        if llm_pid:
525                            role_label += f" — {llm_pid}"
526                        doc.text.addElement(P(text=f"[{role_label}]"))
527                        if content:
528                            for para in content.split("\n\n"):
529                                para = para.strip()
530                                if para:
531                                    doc.text.addElement(P(text=para))
532
533            # Final output
534            output = transcript["output"]
535            if output:
536                doc.text.addElement(H(text="Ergebnis", outlinelevel=2))
537                for para in output.split("\n\n"):
538                    para = para.strip()
539                    if para:
540                        doc.text.addElement(P(text=para))
541
542            # Audit trail
543            if audit_entries:
544                doc.text.addElement(H(text="Audit-Trail", outlinelevel=2))
545                for entry in audit_entries:
546                    llm_pid = entry.get("llm_profile_id", "")
547                    line = f"{entry.get('timestamp', '')} | {entry.get('event_type', '')} | {entry.get('node_id', '')} | {entry.get('actor', '')}"
548                    if llm_pid:
549                        line += f" | LLM: {llm_pid}"
550                    doc.text.addElement(P(text=line))
551
552            doc.save(str(path))
553        except ImportError:
554            # Fallback: write HTML if odfpy not available
555            path = path.with_suffix(".html")
556            path.write_text(html_content, encoding="utf-8")
557            logger.warning("odfpy not available; wrote HTML instead: %s", path)

563    @staticmethod
564    def _render_html(
565        session_id: str,
566        transcript: dict[str, Any],
567        audit_entries: list[dict[str, Any]],
568    ) -> str:
569        esc = html_mod.escape
570        title = esc(transcript["title"] or f"Debatte {session_id[:8]}")
571
572        # --- Metadata table ---
573        meta_pairs = [
574            ("Session-ID", session_id),
575            ("Titel", esc(transcript["title"])),
576            ("Sprache", esc(transcript["language"])),
577            ("Modell", esc(transcript["model"])),
578            ("Max. Runden", str(transcript["max_rounds"])),
579            ("Konsens-Schwelle", f"{transcript['threshold']:.0%}"),
580            ("Status", esc(transcript["status"])),
581            ("Runden absolviert", str(transcript["current_round"])),
582            ("Finaler Konsens", f"{transcript['final_consensus']:.1%}"),
583            ("Generiert", datetime.now().strftime("%Y-%m-%d %H:%M:%S")),
584        ]
585        meta_rows = "".join(f"<tr><td><strong>{k}</strong></td><td>{v}</td></tr>" for k, v in meta_pairs)
586
587        # --- Case description ---
588        case_text = transcript["case_text"]
589        case_section = ""
590        if case_text:
591            case_section = f'<h2>Fallbeschreibung</h2><div class="case-text">{esc(case_text)}</div>'
592
593        # --- Rounds ---
594        rounds_html = ""
595        rounds = transcript["rounds"]
596        if rounds:
597            rounds_html += "<h2>Debatte-Transkript</h2>"
598            for rd in rounds:
599                round_num = rd.get("round", "?")
600                consensus = rd.get("consensus", 0.0)
601                rounds_html += f"<h3>Runde {round_num} — Konsens: {consensus:.1%}</h3>"
602                for ao in rd.get("agent_outputs", []):
603                    role = ao.get("role", "unbekannt")
604                    content = esc(ao.get("content", ""))
605                    tokens = ao.get("tokens_used", 0)
606                    duration_ms = ao.get("duration_ms", 0)
607                    llm_pid = ao.get("llm_profile_id", "")
608                    display_role = esc(_display_agent_role(role))
609                    if llm_pid:
610                        display_role += f' <span style="font-weight:normal;color:#666;">— {esc(llm_pid)}</span>'
611                    rounds_html += f'<div class="agent-block"><div class="agent-role">{display_role}</div><div class="agent-content">{content}</div>'
612                    meta_parts = []
613                    if tokens:

235    @staticmethod
236    def _extract_transactional_protocol(
237        audit_entries: list[dict[str, Any]],
238    ) -> list[dict[str, Any]]:
239        """Extract CriticItem → BuildResponse → PragmatistEvaluation data from audit entries.
240
241        Parses ``builder_iteration`` and ``pragmatist_evaluation`` workflow
242        events from the audit log and merges them into a unified protocol
243        table for the report.
244        """
245        import json as _json
246
247        iterations: dict[int, dict] = {}  # draft_version → builder data
248        evaluations: list[dict] = []
249
250        for entry in audit_entries:
251            event_type = entry.get("event_type", "")
252            raw_output = entry.get("output_content", "")
253
254            if event_type == "builder_iteration":
255                try:
256                    if isinstance(raw_output, str) and raw_output:
257                        data = _json.loads(raw_output)
258                    elif isinstance(raw_output, dict):
259                        data = raw_output
260                    else:
261                        data = {}
262                except (ValueError, TypeError):
263                    data = {}
264                dv = entry.get("draft_version", data.get("draft_version", 0))
265                iterations[dv] = {
266                    "draft_version": dv,
267                    "constructivity_score": entry.get("constructivity_score", data.get("constructivity_score")),
268                    "build_response_count": data.get("build_response_count", 0),
269                }
270
271            elif event_type == "pragmatist_evaluation":
272                try:
273                    if isinstance(raw_output, str) and raw_output:
274                        data = _json.loads(raw_output)
275                    elif isinstance(raw_output, dict):
276                        data = raw_output
277                    else:
278                        data = {}
279                except (ValueError, TypeError):
280                    data = {}
281                verdicts = data.get("verdicts", [])
282                for v in verdicts:
283                    evaluations.append(v)
284
285        # Build merged protocol entries
286        protocol: list[dict[str, Any]] = []
287
288        for ev in evaluations:
289            resp_to = ev.get("response_to", "")
290            verdict = ev.get("verdict", "pending")
291            feasibility = ev.get("feasibility")
292
293            # Find the iteration this belongs to
294            draft_ver = 0
295            for dv, it_data in sorted(iterations.items()):
296                draft_ver = dv
297
298            protocol.append(
299                {
300                    "critic_item_id": resp_to,
301                    "build_response_summary": f"Verdict: {verdict}",
302                    "verdict": verdict,
303                    "feasibility": feasibility,
304                    "draft_version": draft_ver,
305                }
306            )
307
308        # If no pragmatist evaluations, fall back to builder iterations
309        if not protocol and iterations:
310            for dv, it_data in sorted(iterations.items()):
311                protocol.append(
312                    {
313                        "critic_item_id": f"Iteration {dv}",
314                        "build_response_summary": f"{it_data.get('build_response_count', 0)} Antworten",
315                        "verdict": "pending",
316                        "feasibility": it_data.get("constructivity_score"),
317                        "draft_version": dv,
318                    }
319                )
320
321        return protocol
