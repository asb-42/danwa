backend.workflow.node_functions

Node functions for LangGraph workflow execution.

Each function (or factory) produces a partial state update dict that LangGraph merges into the shared WorkflowState. Agent nodes resolve their AgentBlueprint at runtime and call the LLM via LLMService.

 50async def input_node(state: WorkflowState) -> dict:
 51    """Input node — sets the context from the workflow input.
 52
 53    No LLM call.  Simply passes through the user's case text.
 54    """
 55    session_id = state.get("session_id", "")
 56    node_id = state.get("current_node_id", "wf-input")
 57
 58    await publish_async(
 59        session_id,
 60        "node.start",
 61        {
 62            "node_id": node_id,
 63            "node_type": "wf-input",
 64            "round": state.get("current_round", 1),
 65        },
 66    )
 67
 68    output: WorkflowNodeOutput = {
 69        "node_id": node_id,
 70        "node_type": "wf-input",
 71        "role": "input",
 72        "content": state.get("context", ""),
 73        "tokens_used": 0,
 74        "duration_ms": 0,
 75        "status": "completed",
 76    }
 77
 78    await publish_async(
 79        session_id,
 80        "node.complete",
 81        {
 82            "node_id": node_id,
 83            "node_type": "wf-input",
 84            "role": "input",
 85            "content": output["content"][:200],
 86            "tokens_used": 0,
 87            "duration_ms": 0,
 88        },
 89    )
 90
 91    # --- Audit log ---
 92    try:
 93        get_audit_logger().log_node_execution(
 94            session_id=session_id,
 95            workflow_id=state.get("workflow_id", ""),
 96            workflow_version=state.get("workflow_version", 1),
 97            node_id=node_id,
 98            actor="input",
 99            input_data={"context": state.get("context", "")},
100            output_data=output,
101        )
102    except Exception:
103        logger.debug("Audit logging failed for input_node", exc_info=True)
104
105    return {
106        "node_outputs": [output],
107        "current_draft": state.get("context", ""),
108    }

225def agent_node_factory(
226    node_id: str,
227    node_type: str,
228    resolved_config: dict,
229) -> Callable[[WorkflowState], dict]:
230    """Create an agent node function for a workflow node.
231
232    Args:
233        node_id: The workflow node ID (e.g. ``"node_strategist_1"``).
234        node_type: The node type (e.g. ``"wf-strategist"``).
235        resolved_config: Dict with keys ``blueprint_id``, ``blueprint_name``,
236            ``llm_profile_id``, ``llm_model``, ``role_definition_id``,
237            ``role``, ``prompt_template_id``.
238
239    Returns:
240        An async callable that takes ``WorkflowState`` and returns a partial
241        state update dict.
242    """
243    role = resolved_config.get("role", node_type.replace("wf-", ""))
244    llm_profile_id = resolved_config.get("llm_profile_id", "")
245    resolved_config.get("blueprint_name", role)
246
247    # Extract tone_profile_source_node_id from resolved config
248    tone_profile_source_node_id = resolved_config.get("tone_profile_source_node_id")
249
250    async def _agent_node(state: WorkflowState) -> dict:
251        session_id = state.get("session_id", "")
252        current_round = state.get("current_round", 1)
253        start_time = time.monotonic()
254
255        # --- Publish: node started ---
256        await publish_async(
257            session_id,
258            "node.start",
259            {
260                "node_id": node_id,
261                "node_type": node_type,
262                "role": role,
263                "round": current_round,
264            },
265        )
266
267        # --- Build system prompt ---
268        system_prompt = _resolve_system_prompt(resolved_config, state)
269
270        # --- Inject tone profile if configured ---
271        tone_profile_name = None
272        if tone_profile_source_node_id:
273            tone_profiles = state.get("tone_profiles", {})
274            profile_data = tone_profiles.get(tone_profile_source_node_id)
275            if profile_data:
276                try:
277                    from backend.blueprints.models import ToneProfile
278                    from backend.services.tone_prompt_injector import inject_tone_profile
279
280                    profile = ToneProfile.model_validate(profile_data)
281                    system_prompt = inject_tone_profile(system_prompt, profile)
282                    tone_profile_name = profile.name
283                    logger.info(
284                        "Injected tone profile '%s' into agent %s (node %s)",
285                        profile.name,
286                        role,
287                        node_id,
288                    )
289                except Exception as exc:
290                    logger.warning(
291                        "Failed to inject tone profile for agent %s (node %s): %s",
292                        role,
293                        node_id,
294                        exc,
295                    )
296
297        # --- Build user prompt ---
298        context = state.get("context", "")
299        current_draft = state.get("current_draft", "")
300        language = state.get("language", "de")
301
302        user_prompt = f"Case: {context}"
303
304        # Inject RAG context (document content)
305        rag_context = state.get("rag_context", "")
306        if rag_context:
307            user_prompt += f"\n\n--- DOCUMENT CONTEXT ---\n{rag_context}"
308
309        if current_draft:
310            user_prompt += f"\n\nCurrent draft:\n{current_draft}"
311
312        # Inject pending interjections
313        interjections = state.get("interjection_queue", [])
314        if interjections:
315            inj_text = "\n\n--- ADDITIONAL CONTEXT (User) ---\n"
316            inj_text += "\n".join(f"- {inj['content']}" for inj in interjections)
317            user_prompt += inj_text
318
319        if language == "en":
320            user_prompt += "\n\nPlease respond in English."
321        else:
322            user_prompt += "\n\nBitte antworte auf Deutsch."
323
324        # --- LLM call ---
325        content = ""
326        tokens_used = 0
327        duration_ms = 0
328        status = "completed"
329
330        try:
331            llm_service = LLMService(
332                profile_id=llm_profile_id,
333                profile_service=_get_profile_service(),
334            )
335            gen_result = await llm_service.generate(
336                prompt=user_prompt,
337                system_prompt=system_prompt,
338            )
339            content = gen_result.content
340            tokens_used = gen_result.tokens_out if gen_result.tokens_out > 0 else len(content.split())
341            duration_ms = gen_result.duration_ms
342
343            logger.info(
344                "Agent %s (node %s, round %d): LLM response (%d tokens, %dms)",
345                role,
346                node_id,
347                current_round,
348                tokens_used,
349                duration_ms,
350            )
351
352        except Exception as exc:
353            logger.error(
354                "Agent %s (node %s, round %d): LLM call FAILED: %s",
355                role,
356                node_id,
357                current_round,
358                exc,
359                exc_info=True,
360            )
361            content = f"[{role}] Round {current_round}: LLM call failed ({exc})"
362            tokens_used = len(content.split())
363            status = "failed"
364
365        elapsed_ms = int((time.monotonic() - start_time) * 1000)
366
367        output: WorkflowNodeOutput = {
368            "node_id": node_id,
369            "node_type": node_type,
370            "role": role,
371            "content": content,
372            "tokens_used": tokens_used,
373            "duration_ms": elapsed_ms,
374            "status": status,
375        }
376
377        # --- Publish: node completed ---
378        await publish_async(
379            session_id,
380            "node.complete",
381            {
382                "node_id": node_id,
383                "node_type": node_type,
384                "role": role,
385                "content": content[:500],
386                "tokens_used": tokens_used,
387                "duration_ms": elapsed_ms,
388            },
389        )
390
391        # --- Audit log ---
392        try:
393            al = get_audit_logger()
394            audit_metadata = {}
395            if tone_profile_name:
396                audit_metadata["tone_profile_name"] = tone_profile_name
397
398            if status == "failed":
399                al.log_node_failed(
400                    session_id=session_id,
401                    workflow_id=state.get("workflow_id", ""),
402                    workflow_version=state.get("workflow_version", 1),
403                    node_id=node_id,
404                    actor=role,
405                    error=content,
406                )
407            else:
408                al.log_node_execution(
409                    session_id=session_id,
410                    workflow_id=state.get("workflow_id", ""),
411                    workflow_version=state.get("workflow_version", 1),
412                    node_id=node_id,
413                    actor=role,
414                    input_data={
415                        "system_prompt": system_prompt,
416                        "user_prompt": user_prompt,
417                        **audit_metadata,
418                    },
419                    output_data={"content": content, "tokens_used": tokens_used},
420                    llm_profile_id=llm_profile_id,
421                    latency_ms=duration_ms,
422                    prompt_tokens=0,
423                    completion_tokens=tokens_used,
424                )
425        except Exception:
426            logger.debug("Audit logging failed for agent_node %s", node_id, exc_info=True)
427
428        return {
429            "node_outputs": [output],
430            "messages": [{"role": role, "content": content, "round": current_round}],
431            "current_draft": state.get("current_draft", "") + "\n" + content,
432        }
433
434    return _agent_node

437def moderator_node_factory(
438    node_id: str,
439    resolved_config: dict,
440    threshold: float = 0.7,
441) -> Callable[[WorkflowState], dict]:
442    """Create a moderator node that also evaluates consensus.
443
444    Same as ``agent_node_factory`` but additionally computes a consensus
445    score and publishes a ``consensus.reached`` SSE event.
446    """
447    base_fn = agent_node_factory(node_id, "wf-moderator", resolved_config)
448
449    async def _moderator_node(state: WorkflowState) -> dict:
450        result = await base_fn(state)
451
452        # Simple consensus heuristic: based on draft length stability
453        # In a real implementation, this would use LLM-based evaluation
454        current_draft = result.get("current_draft", state.get("current_draft", ""))
455        draft_length = len(current_draft)
456        # Heuristic: longer drafts with more contributions = higher consensus
457        num_outputs = len(state.get("node_outputs", [])) + len(result.get("node_outputs", []))
458        consensus = min(1.0, (num_outputs * 0.15) + (draft_length / 10000))
459
460        session_id = state.get("session_id", "")
461        await publish_async(
462            session_id,
463            "consensus.reached",
464            {
465                "score": round(consensus, 2),
466                "threshold": threshold,
467                "round": state.get("current_round", 1),
468            },
469        )
470
471        result["final_consensus"] = round(consensus, 2)
472        return result
473
474    return _moderator_node

482def gate_node_factory(
483    node_id: str,
484    condition_expr: str = "",
485) -> Callable[[WorkflowState], dict]:
486    """Create a gate node that evaluates a condition.
487
488    The gate node itself doesn't route — routing is handled by
489    ``workflow_routers.route_conditional()``.  This function just
490    publishes SSE events and records the gate evaluation.
491
492    Args:
493        node_id: The workflow node ID.
494        condition_expr: The condition expression (for logging).
495    """
496
497    async def _gate_node(state: WorkflowState) -> dict:
498        session_id = state.get("session_id", "")
499        current_round = state.get("current_round", 1)
500
501        await publish_async(
502            session_id,
503            "node.start",
504            {
505                "node_id": node_id,
506                "node_type": "wf-gate",
507                "round": current_round,
508            },
509        )
510
511        # Evaluate condition for logging
512        condition_result = False
513        try:
514            if condition_expr:
515                condition_result = bool(
516                    eval(condition_expr, {"__builtins__": {}}, dict(state))  # noqa: S307
517                )
518        except Exception as exc:
519            logger.warning("Gate condition '%s' raised: %s", condition_expr, exc)
520
521        output: WorkflowNodeOutput = {
522            "node_id": node_id,
523            "node_type": "wf-gate",
524            "role": "gate",
525            "content": f"Gate evaluated: {condition_expr} → {condition_result}",
526            "tokens_used": 0,
527            "duration_ms": 0,
528            "status": "completed",
529        }
530
531        await publish_async(
532            session_id,
533            "node.complete",
534            {
535                "node_id": node_id,
536                "node_type": "wf-gate",
537                "role": "gate",
538                "content": output["content"],
539                "tokens_used": 0,
540                "duration_ms": 0,
541            },
542        )
543
544        # --- Audit log ---
545        try:
546            get_audit_logger().log_node_execution(
547                session_id=session_id,
548                workflow_id=state.get("workflow_id", ""),
549                workflow_version=state.get("workflow_version", 1),
550                node_id=node_id,
551                actor="gate",
552                input_data={"condition": condition_expr},
553                output_data={"result": condition_result},
554            )
555        except Exception:
556            logger.debug("Audit logging failed for gate_node %s", node_id, exc_info=True)
557
558        return {"node_outputs": [output]}
559
560    return _gate_node

568def tone_profile_node_factory(
569    node_id: str,
570    node_config: dict,
571) -> Callable[[WorkflowState], dict]:
572    """Create a tone profile node function.
573
574    Loads the ToneProfile (from catalog or inline) and writes it into
575    ``state["tone_profiles"][node_id]``.  No LLM call, no output.
576
577    Args:
578        node_id: The workflow node ID.
579        node_config: The node config dict with keys ``tone_profile_id``
580            or ``inline_profile``.
581    """
582    from backend.blueprints.models import ToneProfile
583    from backend.blueprints.repository import BlueprintRepository
584
585    tone_profile_id = node_config.get("tone_profile_id")
586    inline_profile_data = node_config.get("inline_profile")
587
588    async def _tone_profile_node(state: WorkflowState) -> dict:
589        session_id = state.get("session_id", "")
590
591        await publish_async(
592            session_id,
593            "node.start",
594            {
595                "node_id": node_id,
596                "node_type": "wf-tone-profile",
597                "role": "tone-profile",
598            },
599        )
600
601        profile: ToneProfile | None = None
602
603        if tone_profile_id:
604            # Load from catalog
605            try:
606                repo = BlueprintRepository()
607                profile = repo.get_tone_profile(tone_profile_id)
608                if profile is None:
609                    logger.warning("ToneProfile '%s' not found in catalog", tone_profile_id)
610            except Exception as exc:
611                logger.error("Failed to load ToneProfile '%s': %s", tone_profile_id, exc)
612
613        elif inline_profile_data:
614            # Use inline profile
615            try:
616                if isinstance(inline_profile_data, dict):
617                    profile = ToneProfile.model_validate(inline_profile_data)
618                else:
619                    profile = inline_profile_data
620            except Exception as exc:
621                logger.error("Failed to parse inline ToneProfile: %s", exc)
622
623        # Update tone_profiles in state
624        tone_profiles = dict(state.get("tone_profiles", {}))
625        if profile is not None:
626            tone_profiles[node_id] = profile.model_dump()
627
628        output = {
629            "node_id": node_id,
630            "node_type": "wf-tone-profile",
631            "role": "tone-profile",
632            "content": f"Tone profile loaded: {profile.name if profile else 'none'}",
633            "tokens_used": 0,
634            "duration_ms": 0,
635            "status": "completed" if profile else "failed",
636        }
637
638        await publish_async(
639            session_id,
640            "node.complete",
641            {
642                "node_id": node_id,
643                "node_type": "wf-tone-profile",
644                "role": "tone-profile",
645                "content": output["content"],
646                "tokens_used": 0,
647                "duration_ms": 0,
648            },
649        )
650
651        # --- Audit log ---
652        try:
653            get_audit_logger().log_node_execution(
654                session_id=session_id,
655                workflow_id=state.get("workflow_id", ""),
656                workflow_version=state.get("workflow_version", 1),
657                node_id=node_id,
658                actor="tone-profile",
659                input_data={"tone_profile_id": tone_profile_id},
660                output_data={"profile_name": profile.name if profile else None},
661            )
662        except Exception:
663            logger.debug("Audit logging failed for tone_profile_node %s", node_id, exc_info=True)
664
665        return {
666            "tone_profiles": tone_profiles,
667            "node_outputs": [output],
668        }
669
670    return _tone_profile_node

678async def interjection_node(state: WorkflowState) -> dict:
679    """Interjection node — consumes queued user input or pauses.
680
681    If there are pending interjections in the queue, consumes them and
682    appends to the state.  If the queue is empty, sets ``is_paused=True``
683    to signal the execution engine to pause and wait for user input.
684    """
685    session_id = state.get("session_id", "")
686    node_id = state.get("current_node_id", "wf-user-injection")
687
688    await publish_async(
689        session_id,
690        "node.start",
691        {
692            "node_id": node_id,
693            "node_type": "wf-user-injection",
694            "round": state.get("current_round", 1),
695        },
696    )
697
698    queue = state.get("interjection_queue", [])
699
700    if queue:
701        # Consume all pending interjections
702        consumed_ids = [item.get("id", "") for item in queue]
703        combined_content = "\n".join(item.get("content", "") for item in queue)
704
705        output: WorkflowNodeOutput = {
706            "node_id": node_id,
707            "node_type": "wf-user-injection",
708            "role": "user-injection",
709            "content": combined_content,
710            "tokens_used": 0,
711            "duration_ms": 0,
712            "status": "completed",
713        }
714
715        await publish_async(
716            session_id,
717            "node.complete",
718            {
719                "node_id": node_id,
720                "node_type": "wf-user-injection",
721                "role": "user-injection",
722                "content": combined_content[:500],
723                "tokens_used": 0,
724                "duration_ms": 0,
725            },
726        )
727
728        # --- Audit log ---
729        try:
730            get_audit_logger().log_node_execution(
731                session_id=session_id,
732                workflow_id=state.get("workflow_id", ""),
733                workflow_version=state.get("workflow_version", 1),
734                node_id=node_id,
