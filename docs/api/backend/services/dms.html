173    def move_document_to(self, document_id: str, target_dms: "DMS", target_project_id: str) -> bool:
174        """Move a document to another project's DMS.
175
176        Reads the document and its chunks from this DMS, inserts them into
177        the target DMS (with a new document ID), re-indexes in the target
178        vector store, then deletes from this DMS.
179
180        Returns True on success. If the target insert fails, the source
181        document is left unchanged. If the source delete fails after a
182        successful target insert, a warning is logged (document exists in
183        both projects).
184        """
185        doc = self.db.get_document(document_id)
186        if not doc:
187            logger.error("Document %s not found in source DMS", document_id)
188            return False
189
190        chunks = self.db.list_chunks(document_id)
191        chunk_texts = [c["text"] for c in chunks]
192
193        # 1. Insert into target DMS DB (new doc_id is auto-generated by add_document)
194        try:
195            new_doc = target_dms.db.add_document(
196                project_id=target_project_id,
197                filename=doc.get("filename", "unknown"),
198                file_path="",
199                file_type=doc.get("file_type", ""),
200                file_size=doc.get("file_size", 0),
201                original_filename=doc.get("original_filename", ""),
202                page_count=doc.get("page_count", 0),
203                word_count=doc.get("word_count", 0),
204                char_count=doc.get("char_count", 0),
205                ocr_used=bool(doc.get("ocr_used", 0)),
206            )
207            new_doc_id = new_doc["id"]
208            logger.info("Created document entry %s in target project %s", new_doc_id, target_project_id)
209        except Exception as e:
210            logger.error("Failed to create document in target DMS: %s", e)
211            return False
212
213        # 2. Re-index chunks in target DMS
214        try:
215            if chunk_texts:
216                chunk_dicts = [{"text": t, "chunk_index": i, "page": 0} for i, t in enumerate(chunk_texts)]
217                target_dms.vector_store.add_chunks(
218                    document_id=new_doc_id,
219                    chunks=chunk_dicts,
220                    project_id=target_project_id,
221                )
222                metadata = str(
223                    {
224                        "file_name": doc["filename"],
225                        "upload_date": doc.get("uploaded_at", ""),
226                        "project_id": target_project_id,
227                    }
228                )
229                target_dms.db.conn.executemany(
230                    """INSERT INTO document_chunks
231                    (id, document_id, chunk_index, text, embedding_id, page, metadata_json)
232                    VALUES (?, ?, ?, ?, ?, ?, ?)""",
233                    [(str(uuid.uuid4())[:8], new_doc_id, i, ct, "", 0, metadata) for i, ct in enumerate(chunk_texts)],
234                )
235                target_dms.db.conn.commit()
236                logger.info("Indexed %d chunks for document %s in target project", len(chunk_texts), new_doc_id)
237        except Exception as e:
238            logger.error("Failed to index chunks in target DMS, cleaning up target document %s: %s", new_doc_id, e)
239            target_dms.delete_document(new_doc_id)
240            return False
241
242        # 3. Delete from source DMS
243        try:
244            self.delete_document(document_id)
245        except Exception as e:
246            logger.warning(
247                "Document %s copied to target as %s but source deletion failed: %s. Document exists in both projects.",
248                document_id,
249                new_doc_id,
250                e,
251            )
252            return True
253
254        logger.info(
255            "Moved document %s (%s) from project %s to project %s as %s",
256            document_id,
257            doc.get("filename"),
258            doc["project_id"],
259            target_project_id,
260            new_doc_id,
261        )
262        return True

383    def get_manual_rag_context(self, k: int = 5) -> list[dict[str, Any]]:
384        """Get chunks from manually selected RAG documents.
385
386        Uses round-robin sampling across documents to avoid biasing
387        toward the first document when multiple are selected.
388        """
389        try:
390            doc_chunks: list[list[dict]] = []
391            for doc_id in self._manual_rag_docs:
392                chunks = self.metadata_index.get_chunks_by_document(doc_id)
393                if chunks:
394                    doc_chunks.append(chunks)
395
396            if not doc_chunks:
397                return []
398
399            # Round-robin: take one chunk from each document in turn
400            result: list[dict] = []
401            max_len = max(len(c) for c in doc_chunks)
402            for i in range(max_len):
403                for dc in doc_chunks:
404                    if i < len(dc) and len(result) < k:
405                        result.append(dc[i])
406                if len(result) >= k:
407                    break
408
409            return result
410        except Exception as e:
411            logger.error("Failed to get manual RAG context: %s", e)
412            return []

125        if not chunks:

128            corpus = [chunk["text"] for chunk in chunks]

133            top_indices = sorted(range(len(scores)), key=lambda i: scores[i], reverse=True)[:top_n]

140                        "text": chunk["text"],
141                        "metadata": chunk["metadata"],
142                        "bm25_score": scores[idx],

17    def __init__(self, chunk_size: int = 512, overlap: int = 51):
18        self.encoder = tiktoken.get_encoding("cl100k_base")
19        self.chunk_size = chunk_size
20        self.overlap = overlap
21
22    def chunk(self, text: str) -> list[str]:
23        if not text:
24            return []
25        tokens = self.encoder.encode(text)
26        if len(tokens) <= self.chunk_size:
27            return [text]
28        chunks = []
29        start = 0
30        while start < len(tokens):
31            end = min(start + self.chunk_size, len(tokens))
32            chunk_tokens = tokens[start:end]
33            chunks.append(self.encoder.decode(chunk_tokens))
34            start += self.chunk_size - self.overlap
35            if start >= len(tokens):
36                break
37        return chunks
