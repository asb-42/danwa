backend.services.ui_translation_service

UI-Translation Service — Verwaltet Frontend-String-Übersetzungen.

Dieser Service speichert und verwaltet UI-Übersetzungen (i18n) in einer SQLite-Datenbank, ergänzt durch LLM-basierte Ad-hoc-Übersetzung.

Im Gegensatz zum TranslationService (der für Modulinhalte/Blueprints zuständig ist) handelt dieser Service ausschließlich UI-Strings wie Menüs, Buttons, Labels etc.

201    def set_translation(self, key: str, locale: str, value: str, namespace: str = "global", source: str = "manual") -> None:
202        """Speichere oder aktualisiere eine Übersetzung."""
203        now = datetime.now(UTC).isoformat()
204        conn = self._get_conn()
205        try:
206            conn.execute(
207                """
208                INSERT INTO ui_translations (key, locale, value, namespace, source,
209                                             confidence, version, created_at, updated_at)
210                VALUES (?, ?, ?, ?, ?, NULL, 1, ?, ?)
211                ON CONFLICT(key, locale, namespace) DO UPDATE SET
212                    value = excluded.value,
213                    source = excluded.source,
214                    version = ui_translations.version + 1,
215                    updated_at = excluded.updated_at
216            """,
217                (key, locale, value, namespace, source, now, now),
218            )
219            conn.commit()
220            # Cache invalidieren
221            self._locales_cache.pop(locale, None)
222            logger.debug("UI-Übersetzung gespeichert: %s [%s/%s]", key, locale, namespace)
223        finally:
224            conn.close()

226    def get_translation(self, key: str, locale: str, namespace: str = "global") -> str | None:
227        """Hole eine einzelne Übersetzung. Fallback: bundled loaders für EN."""
228        conn = self._get_conn()
229        row = conn.execute("SELECT value FROM ui_translations WHERE key = ? AND locale = ? AND namespace = ?", (key, locale, namespace)).fetchone()
230        conn.close()
231        if row:
232            return row["value"]
233        # Fallback: bundled loader (especially for English source strings)
234        if locale == "en" and namespace == "global":
235            bundled = self._scan_bundled_loaders()
236            return bundled.get("en", {}).get(key)
237        return None

239    def get_translations_bulk(self, locale: str, namespace: str = "global", keys: list[str] | None = None) -> dict[str, str]:
240        """Hole mehrere Übersetzungen auf einmal. Befüllt den lokalen Cache."""
241        conn = self._get_conn()
242        if keys:
243            placeholders = ",".join("?" for _ in keys)
244            query = f"SELECT key, value FROM ui_translations WHERE locale = ? AND namespace = ? AND key IN ({placeholders})"
245            params: list[Any] = [locale, namespace] + keys
246        else:
247            query = "SELECT key, value FROM ui_translations WHERE locale = ? AND namespace = ?"
248            params = [locale, namespace]
249
250        rows = conn.execute(query, params).fetchall()
251        conn.close()
252        result = {r["key"]: r["value"] for r in rows}
253        # Cache befüllen
254        if locale not in self._locales_cache:
255            self._locales_cache[locale] = {}
256        if keys is None:
257            self._locales_cache[locale] = dict(result)
258        else:
259            self._locales_cache[locale].update(result)
260        return result

262    def resolve_bulk_for_locale(self, locale: str, prefix: str = "langpack:") -> dict[str, str]:
263        """Resolve all translations for a locale from namespaces matching a prefix.
264
265        Merges strings from ALL matching namespaces, with later namespaces
266        overriding earlier ones (alphabetical order).
267
268        Args:
269            locale: Target locale code (e.g. 'de', 'de-custom')
270            prefix: Namespace prefix to match (e.g. 'langpack:')
271
272        Returns:
273            Merged dict of key-value pairs
274        """
275        conn = self._get_conn()
276        rows = conn.execute(
277            "SELECT namespace, key, value FROM ui_translations WHERE locale = ? AND namespace LIKE ?",
278            (locale, f"{prefix}%"),
279        ).fetchall()
280        conn.close()
281
282        # Group by namespace, then merge in alphabetical order
283        by_namespace: dict[str, dict[str, str]] = {}
284        for row in rows:
285            ns = row["namespace"]
286            if ns not in by_namespace:
287                by_namespace[ns] = {}
288            by_namespace[ns][row["key"]] = row["value"]
289
290        merged: dict[str, str] = {}
291        for ns in sorted(by_namespace.keys()):
292            merged.update(by_namespace[ns])
293        return merged

297        conn = self._get_conn()

299            "SELECT DISTINCT key FROM ui_translations WHERE namespace = ?",

322    def bulk_import(self, translations: dict[str, dict[str, str]], namespace: str = "global", source: str = "bulk_imported") -> int:
323        """
324        Importiere mehrere Übersetzungen auf einmal.

338    def resolve(self, key: str, locale: str, namespace: str = "global") -> str:
339        """
340        Resolviere eine Übersetzung mit Fallback-Kette:
341        locale → DEFAULT_LOCALE → en → key
342        """
343        chain = [locale, "de", "en"]
344        for loc in chain:
345            val = self.get_translation(key, loc, namespace)
346            if val is not None:
347                return val
348        return key  # Letzter Fallback

350    def resolve_bulk(self, locale: str, namespace: str = "global", keys: list[str] | None = None) -> dict[str, str]:
351        """Bulk-Resolution mit Fallback-Kette."""
352        if keys is None:
353            keys = self.get_all_keys(namespace)
354
355        result = {}
356        primary = self.get_translations_bulk(locale, namespace, keys)
357        fallback_de = self.get_translations_bulk("de", namespace, [k for k in keys if k not in primary]) if locale != "de" else {}
358        fallback_en = self.get_translations_bulk("en", namespace, [k for k in keys if k not in primary and k not in fallback_de])
359
360        result.update(primary)
361        result.update(fallback_de)
362        result.update(fallback_en)
363
364        # Letzter Fallback: Key selbst
365        for k in keys:
366            if k not in result:
367                result[k] = k
368
369        return result

392    def _scan_bundled_loaders(self) -> dict[str, dict[str, str]]:
393        """Scan frontend loader JS files and extract key→value dicts."""
394        root = Path(__file__).resolve().parent.parent.parent
395        loader_dir = root / "frontend" / "src" / "lib" / "i18n" / "loaders"
396        if not loader_dir.exists():
397            return {}
398
399        result = {}
400        for js_file in loader_dir.glob("*.js"):
401            locale = js_file.stem
402            content = js_file.read_text(encoding="utf-8")
403            keys = {}
404            pattern = r"^\s*['\"]([^'\"]+)['\"]\s*:\s*['\"]((?:[^'\"\\]|\\.)*)['\"]"
405            for match in __import__("re").finditer(pattern, content, __import__("re").MULTILINE):
406                keys[match.group(1)] = match.group(2)
407            if keys:
408                result[locale] = keys
409        return result

411    def get_stats(self, namespace: str = "global") -> dict[str, Any]:
412        conn = self._get_conn()
413        rows = conn.execute(
414            """
415            SELECT locale, COUNT(*) as total,
416                    SUM(CASE WHEN source = 'manual' THEN 1 ELSE 0 END) as manual,
417                    SUM(CASE WHEN source = 'bulk_imported' THEN 1 ELSE 0 END) as bulk,
418                    SUM(CASE WHEN source = 'llm_generated' THEN 1 ELSE 0 END) as llm,
419                    MAX(updated_at) as last_updated
420            FROM ui_translations
421            WHERE namespace = ?
422            GROUP BY locale
423        """,
424            (namespace,),
425        ).fetchall()
426        db_stats = {r["locale"]: dict(r) for r in rows}
427        conn.close()
428
429        bundled = self._scan_bundled_loaders()
430        for locale, keys in bundled.items():
431            if locale not in db_stats:
432                db_stats[locale] = {
433                    "locale": locale,
434                    "total": len(keys),
435                    "manual": 0,
436                    "bulk": 0,
437                    "llm": 0,
438                    "last_updated": None,
439                }
440            else:
441                db_stats[locale]["total"] = max(db_stats[locale]["total"], len(keys))
442
443        for locale, s in db_stats.items():
444            s["translated"] = (s.get("llm") or 0) + (s.get("manual") or 0) + (s.get("bulk") or 0)
445
446        return db_stats

448    def get_coverage(self, namespace: str = "global") -> dict[str, Any]:
449        bundled = self._scan_bundled_loaders()
450        en_keys = set(self.get_all_keys(namespace))
451        if not en_keys:
452            en_keys = set(bundled.get("en", {}).keys())
453        if not en_keys:
454            return {}
455
456        result = {}
457        for locale in DEFAULT_LOCALES:
458            if locale in bundled:
459                bundle_keys = set(bundled[locale].keys())
460                translated = len(bundle_keys & en_keys)
461            else:
462                db_keys = set(self.get_translations_bulk(locale, namespace).keys())
463                translated = len(db_keys & en_keys)
464            coverage = translated / len(en_keys) * 100
465            result[locale] = {
466                "total_keys": len(en_keys),
467                "translated": translated,
468                "coverage_pct": round(coverage, 1),
469            }
470
471        conn = self._get_conn()
472        custom_rows = conn.execute(
473            "SELECT DISTINCT locale FROM ui_translations WHERE locale NOT IN ({}) AND namespace = ?".format(",".join("?" for _ in DEFAULT_LOCALES)),
474            (*DEFAULT_LOCALES, namespace),
475        ).fetchall()
476        for row in custom_rows:
477            locale = row["locale"]
478            db_keys = set(self.get_translations_bulk(locale, namespace).keys())
479            translated = len(db_keys & en_keys)
480            coverage = translated / len(en_keys) * 100
481            result[locale] = {
482                "total_keys": len(en_keys),
483                "translated": translated,
484                "coverage_pct": round(coverage, 1),
485            }
486        conn.close()

494    def translate_via_llm(
495        self,
496        key: str,
497        source_text: str,
498        target_locale: str,
499        llm_profile_id: str | None = None,
500        llm=None,
501        job: TranslationJob | None = None,
502    ) -> str:
503        from backend.services.llm_service import LLMService
504        from backend.services.profile_service import ProfileService
505
506        if llm is None:
507            ps = ProfileService()
508            if llm_profile_id is None:
509                llm_profile_id = self._select_llm_for_locale(target_locale, ps)
510            llm = LLMService(profile_id=llm_profile_id, profile_service=ps)
511
512        system_prompt = (
513            f"You are a professional UI/UX translator. "
514            f"Translate the following English UI string to {self._locale_name(target_locale)}. "
515            f"Keep it concise, precise, and consistent with UI conventions. "
516            f"Do NOT translate placeholder variables like {{param}}. "
517            f"Output ONLY the translation, nothing else."
518        )
519
520        try:
521            result = llm.generate_sync(
522                prompt=source_text,
523                system_prompt=system_prompt,
524                temperature=0.2,
525                max_tokens=max(30, len(source_text) // 4),
526            )
527            translated = result.content.strip()
528            self.set_translation(key, target_locale, translated, source="llm_generated")
529            if job:
530                job.completed_strings += 1
531                job.current_key = key
532                job.current_locale = target_locale
533            return translated
534        except Exception as exc:
535            error_msg = str(exc)
536            is_rate_limit = "rate limit" in error_msg.lower() or "429" in error_msg
537            if is_rate_limit:
538                logger.warning("Rate limit hit for %s → %s: %s", key, target_locale, error_msg[:120])
539                raise  # Re-raise rate limit errors so the job can abort early
540            logger.error("LLM-Übersetzung fehlgeschlagen für %s → %s: %s", key, target_locale, error_msg[:200])
541            return source_text

543    def bulk_translate(self, target_locales: list[str] | None = None, namespace: str = "global") -> dict[str, Any]:
544        """Übersetze alle fehlenden Strings per LLM.
545
546        Liest alle bekannten Keys aus der englischen Datei und übersetzt
547        fehlende Strings in die Zielsprachen.
548        """
549        if target_locales is None:
550            target_locales = [loc for loc in DEFAULT_LOCALES if loc not in ("de", "en")]
551
552        all_keys = self.get_all_keys(namespace)
553        results = {}
554
555        llm_profile = settings.service_llm_profile_id or "(not set — using fallback)"
556        logger.info("Bulk-Translation gestartet: locales=%s, LLM=%s", target_locales, llm_profile)
557
558        for locale in target_locales:
559            existing = self.get_translations_bulk(locale, namespace)
560            missing = [k for k in all_keys if k not in existing or not existing[k]]
561
562            if not missing:
563                results[locale] = {"status": "complete", "translated": 0}
564                continue
565
566            logger.info("Übersetze %d Strings nach %s …", len(missing), locale)
567            translated_count = 0
568            for key in missing:
569                en_val = self.get_translation(key, "en", namespace)
570                if en_val:
571                    self.translate_via_llm(key, en_val, locale)
572                    translated_count += 1
573
574            logger.info("Fertig für %s: %d Strings übersetzt", locale, translated_count)
575            results[locale] = {"status": "ok", "translated": translated_count, "total_missing": len(missing)}
576
577        return results
