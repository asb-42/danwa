 244def build_agent_profile_from_bundles(bundle_ids: list[str]) -> list[dict]:
 245    """Build agent_profile list from AgentBundle IDs.
 246
 247    Each bundle resolves to an agent config with its LLM profile and role type.
 248
 249    Args:
 250        bundle_ids: List of AgentBundle IDs.
 251
 252    Returns:
 253        List of agent config dicts: [{role, llm_profile, temperature, bundle_id}, ...]
 254    """
 255    from backend.blueprints.repository import BlueprintRepository
 256    from backend.blueprints.resolver import BundleResolver
 257
 258    repo = BlueprintRepository()
 259    resolver = BundleResolver(repo)
 260    profile = []
 261
 262    for bid in bundle_ids:
 263        try:
 264            bundle = repo.get_bundle(bid)
 265            if not bundle:
 266                logger.warning("Bundle '%s' not found, skipping", bid)
 267                continue
 268            resolved = resolver.resolve(bundle)
 269            profile.append({
 270                "role": resolved.role_type.id,
 271                "llm_profile": resolved.llm_profile.id,
 272                "temperature": resolved.llm_profile.temperature,
 273                "bundle_id": bid,
 274                "system_prompt": resolved.system_prompt,
 275            })
 276        except Exception as exc:
 277            logger.warning("Failed to resolve bundle '%s': %s", bid, exc)
 278
 279    return profile

 287def resolve_rag_context(
 288    project_id: str,
 289    case_text: str,
 290    document_ids: list[str] | None = None,
 291    rag_auto_retrieve: bool = False,
 292    include_debate_results: bool = False,
 293    project_store: ProjectStore | None = None,
 294    store: DebateStore | None = None,
 295) -> tuple[str, int]:
 296    """Resolve RAG context for a debate.
 297
 298    Returns (rag_context_string, document_count).
 299    """
 300    from backend.services.dms.service import get_dms_for_project
 301
 302    try:
 303        dms = get_dms_for_project(project_id, project_store)
 304    except Exception as exc:
 305        logger.warning("Could not initialize DMS for project %s: %s", project_id, exc)
 306        return "", 0
 307
 308    all_chunks: list[dict] = []
 309
 310    if document_ids:
 311        for doc_id in document_ids:
 312            try:
 313                chunks = dms.metadata_index.get_chunks_by_document(doc_id)
 314                all_chunks.extend(chunks)
 315            except Exception as exc:
 316                logger.warning("Failed to get chunks for document %s: %s", doc_id, exc)
 317
 318    if rag_auto_retrieve and case_text:
 319        try:
 320            auto_chunks = dms.auto_retrieve_for_topic(case_text, project_id=project_id, k=10)
 321            all_chunks.extend(auto_chunks)
 322        except Exception as exc:
 323            logger.warning("Auto-retrieve failed for project %s: %s", project_id, exc)
 324
 325    if include_debate_results:
 326        try:
 327            ps = project_store or ProjectStore()
 328            project_dir = ps.get_project_dir(project_id)
 329            proj_store = DebateStore(data_dir=project_dir / "debates")
 330
 331            debates = proj_store.list_all(limit=50)
 332            debate_count = 0
 333
 334            for d in debates:
 335                if d.get("status") in ("completed",) and d.get("debate_id"):
 336                    transcript = _build_transcript_for_followup(d)
 337                    summary = _generate_rag_friendly_summary(transcript)
 338                    from backend.services.dms.chunker import TextChunker
 339
 340                    chunker = TextChunker()
 341                    chunks = chunker.chunk_text(summary, f"debate_result_{d.get('debate_id', '')[:8]}")
 342                    all_chunks.extend(chunks[:3])
 343
 344                    debate_count += 1
 345                    if debate_count >= 5:
 346                        break
 347        except Exception as exc:
 348            logger.warning("Failed to include debate results in RAG context: %s", exc)
 349
 350    if not all_chunks:
 351        return "", 0
 352
 353    seen_texts: set[str] = set()
 354    unique_chunks: list[dict] = []
 355    for chunk in all_chunks:
 356        text = chunk.get("text", "")
 357        if text and text not in seen_texts:
 358            seen_texts.add(text)
 359            unique_chunks.append(chunk)
 360
 361    if document_ids:
 362        rag_context = dms.format_rag_context(unique_chunks, max_chars=200_000)
 363    else:
 364        rag_context = dms.format_rag_context(unique_chunks)
 365    doc_count = len(document_ids) if document_ids else 0
 366
 367    logger.info(
 368        "RAG context resolved for project %s: %d unique chunks from %d documents",
 369        project_id,
 370        len(unique_chunks),
 371        doc_count,
 372    )
 373    return rag_context, doc_count

 459def _select_service_llm(profile_service) -> str:
 460    """Select the best service LLM profile for system/background tasks.
 461
 462    Strategy:
 463    1. Use configured settings.service_llm_profile_id if eligible.
 464    2. Otherwise find first eligible profile preferring openrouter + large context.
 465    3. Fall back to first available text-LLM.
 466    """
 467    try:
 468        if settings.service_llm_profile_id:
 469            preferred = profile_service.get_llm_profile(settings.service_llm_profile_id)
 470            if preferred and is_service_llm_eligible(preferred)[0]:
 471                return settings.service_llm_profile_id
 472    except Exception:
 473        pass
 474    try:
 475        all_profiles = profile_service.list_llm_profiles()
 476        eligible = [p for p in all_profiles if is_service_llm_eligible(p)[0]]
 477        if eligible:
 478            eligible.sort(key=lambda p_: (0 if p_.provider.value == "openrouter" else 1, -(p_.context_window or 0)))
 479            return eligible[0].id
 480    except Exception:
 481        pass
 482    # Last resort: return first text-LLM
 483    try:
 484        all_profiles = profile_service.list_llm_profiles()
 485        text_profiles = [p for p in all_profiles if getattr(p, "profile_type", "text") == "text"]
 486        if text_profiles:
 487            return text_profiles[0].id
 488    except Exception:
 489        pass
 490    raise RuntimeError("No suitable LLM profile available for service tasks")

 493def validate_title(title: str, case_text: str) -> tuple[bool, str]:
 494    """Check whether a generated title is acceptable.
 495
 496    Returns:
 497        (valid: bool, reason: str) — reason explains why validation failed.
 498    """
 499    import re as _re
 500
 501    if len(title) < 10:
 502        return False, "Zu kurz (< 10 Zeichen)"
 503    if len(title) > 150:
 504        return False, "Zu lang (> 150 Zeichen)"
 505    if title.lower()[:15] in case_text.lower()[:50]:
 506        return False, "Identisch mit Fallbeschreibung"
 507    meta_patterns = [
 508        "the user",
 509        "the case",
 510        "this debate",
 511        "würde gerne",
 512        "der benutzer",
 513        "der fall",
 514        "diese debatte",
 515        "hier ist",
 516        "here is",
 517        "i suggest",
 518        "ich schlage",
 519    ]
 520    if any(p in title.lower() for p in meta_patterns):
 521        return False, "Enthält Meta-Text"
 522    if _re.match(r"^[\s\"'„" r"''`\-–:.,!?]+$", title):
 523        return False, "Nur Sonderzeichen"
 524    return True, "OK"

 527def _post_process_title(raw: str, case_text: str) -> str:
 528    """Clean and validate a generated title, falling back if needed."""
 529    _re = __import__("re")
 530    verbose = [
 531        r"^Here(?:'s| is) (?:a |the |an )?(?:suggested |proposed )?(?:debate )?title[:\s]*",
 532        r"^The title is[:\s]*",
 533        r"^Title[:\s]*",
 534        r"^I suggest[:\s]*",
 535        r"^I propose[:\s]*",
 536        r"^I think[:\s]*",
 537        r"^Der Titel lautet[:\s]*",
 538        r"^Titel[:\s]*",
 539        r"^Hier ist ein Titel[:\s]*",
 540        r"^Ich schlage vor[:\s]*",
 541        r"^Ein passender Titel (?:wäre|ist)[:\s]*",
 542    ]
 543    cleaned = raw.strip()
 544    for pat in verbose:
 545        cleaned = _re.sub(pat, "", cleaned, count=1, flags=_re.IGNORECASE).strip()
 546    for ch in ['"', "'", "„", """, """, """, """, "`", "-", "–", ":", ".", ","]:
 547        cleaned = cleaned.strip(ch)
 548    cleaned = cleaned.strip()
 549
 550    if not cleaned or len(cleaned) < 15 or len(cleaned) > 150:
 551        return _fallback_title(case_text)
 552
 553    reflection = [
 554        r"\b(?:the user|der benutzer)\b",
 555        r"\bbased on\b",
 556        r"\bbasierend auf\b",
 557        r"\b(?:description|beschreibung)\b.*\b(?:about|um)\b",
 558        r"^-\s",
 559    ]
 560    if any(_re.search(p, cleaned, _re.IGNORECASE) for p in reflection):
 561        return _fallback_title(case_text)
 562
 563    # Final validation using standalone validate_title()
 564    valid, reason = validate_title(cleaned, case_text)
 565    if not valid:
 566        logger.debug("Title validation failed: %s — %s", reason, cleaned[:80])
 567        return _fallback_title(case_text)
 568
 569    return cleaned

 572def _fallback_title(case_text: str) -> str:
 573    """Fallback: use beginning of case text as title."""
 574    fallback = case_text[:120].strip()
 575    for sep in [".", "?", "!"]:
 576        if sep in fallback:
 577            fallback = fallback[: fallback.rfind(sep) + 1]
 578            break
 579    if len(fallback) > 150:
 580        fallback = fallback[:147] + "..."
 581    return fallback

 853def build_followup_case(debate_id: str, focus_topic: str | None = None, store: DebateStore | None = None) -> str:
 854    """Baut einen neuen case_text aus den Ergebnissen der Vordebatte (P0)."""

 857        return focus_topic or "Fortsetzung einer vorherigen Debatte."

 859    transcript = _build_transcript_for_followup(debate)

 865            if output.get("role") in ("strategist", "moderator", "critic", "optimizer"):
 866                content = output.get("content", "")

 869
 870    consensus = transcript.get("final_consensus", 0.0)
 871    current_round = transcript.get("current_round", 0)
 872    title = debate.get("title", "unbenannte Debatte")

 878        f"Konsensgrad von {consensus * 100:.0f}% abgeschlossen.\n\n"

 885def build_followup_prompt(previous_debate: dict, new_topic: str) -> str:
 886    """Erstellt einen strukturierten Prompt mit Rollen-Anweisungen (P1)."""
 887
 888    strongest: dict[str, list[str]] = {}
 889    for round_data in previous_debate.get("rounds", []):
 890        for output in round_data.get("agent_outputs", []):
 891            role = output.get("role", "unknown")
 892            if role not in strongest:
 893                strongest[role] = []
 894            content = output.get("content", "")
 895            if content:
 896                strongest[role].append(content[:200])
 897
 898    prompt_parts = [
 899        f"""Du bist Teil eines Multi-Agenten-Debattensystems.
 900
 901## Kontext
 902Eine vorherige Debatte zum Thema "{previous_debate.get("title", "")}"
 903wurde nach {previous_debate.get("current_round", "?")} Runden mit einem
 904Konsensgrad von {previous_debate.get("final_consensus", 0) * 100:.0f}% abgeschlossen.
 905""",
 906    ]
 907
 908    for role, contents in strongest.items():
 909        prompt_parts.append(f"\n### {role.upper()} — Wichtigste Argumente:\n")
 910        for c in contents[-3:]:
 911            prompt_parts.append(f"- {c}")
 912
 913    prompt_parts.extend(
 914        [
 915            "\n## Neue Aufgabe",
 916            f"Diskutiere nun: **{new_topic}**",
 917            "",
 918            "Richtlinien:",
 919            "1. Baue auf den vorherigen Erkenntnissen auf",
 920            "2. Widerlege oder bestätige frühere Schlussfolgerungen",
 921            "3. Bringe neue Perspektiven ein, die in der Origin-Debatte fehlten",
 922            "",
 923            "Halte den Prompt unter 4.000 Tokens.",
 924        ]
 925    )
 926
 927    return "\n".join(prompt_parts)

 948def create_fork_debate(
 949    original_debate_id: str,
 950    new_title: str,
 951    fork_from_round: int | None,
 952    fork_reason: str | None,
 953    modified_personas: dict[str, str] | None,
 954    modified_prompt_variant: str | None,
 955    store: DebateStore,
 956    inherit_personas: bool = True,
 957    inherit_llm_profile: bool = True,
 958) -> dict:
 959    """Erstellt eine Deep-Copy einer Debatte als Fork (P4)."""
 960    import copy
 961    import uuid
 962    from datetime import UTC, datetime
 963
 964    original = store.get(original_debate_id)
 965    if not original:
 966        raise ValueError(f"Original debate {original_debate_id} not found")
 967
 968    new_id = str(uuid.uuid4())
 969    now = datetime.now(UTC)
 970
 971    # Deep-Copy des Original-Debatte-Dicts
 972    fork = copy.deepcopy(original)
 973    fork["debate_id"] = new_id
 974    fork["title"] = new_title
 975    fork["status"] = "pending"
 976    fork["created_at"] = now
 977    fork["updated_at"] = now
 978    fork["current_round"] = 0
 979
 980    # Fork-Metadaten setzen
 981    fork["fork_info"] = {
 982        "parent_debate_id": original_debate_id,
 983        "fork_round": fork_from_round,
 984        "fork_reason": fork_reason,
 985    }
 986
 987    # Runden nach fork_from_round abschneiden
 988    if fork_from_round is not None and fork_from_round >= 0:
 989        fork["rounds"] = [r for r in fork.get("rounds", []) if r.get("round", 0) <= fork_from_round]
 990
 991    # Agent-Personas anpassen
 992    if modified_personas and isinstance(fork.get("request"), dict):
 993        original_personas = fork["request"].get("agent_persona_ids", {})
 994        original_personas.update(modified_personas)
 995        fork["request"]["agent_persona_ids"] = original_personas
 996    elif modified_personas and hasattr(fork.get("request", None), "agent_persona_ids"):
 997        req = fork["request"]
 998        for role, persona_id in modified_personas.items():
 999            setattr(req, "agent_persona_ids", {**req.agent_persona_ids, role: persona_id})
1000
1001    # Prompt-Variante anpassen
1002    if modified_prompt_variant and isinstance(fork.get("request"), dict):
1003        fork["request"]["prompt_variant"] = modified_prompt_variant
1004    elif modified_prompt_variant and hasattr(fork.get("request", None), "prompt_variant"):
1005        fork["request"].prompt_variant = modified_prompt_variant
1006
1007    # Vorbefüllung des case_text mit Kontext aus der Original-Debatte
1008    if isinstance(fork.get("request"), dict):
1009        original_case = fork["request"].get("case", {})
1010        if isinstance(original_case, dict):
1011            original_text = original_case.get("text", "")
1012        else:
1013            original_text = str(original_case)
1014        fork["request"]["case"] = {"text": original_text}
1015    elif hasattr(fork.get("request", None), "case"):
1016        # Fallback: bereits vorhanden
1017        pass
1018
1019    # Fork-Historie im debate_json pflegen
1020    existing_forks = original.get("fork_history", [])
1021    fork["fork_history"] = existing_forks + [
1022        {
1023            "parent_id": original_debate_id,
1024            "fork_round": fork_from_round,
1025            "reason": fork_reason,
1026        }
1027    ]
1028
1029    store.put(new_id, fork)
1030    return {
1031        "debate_id": new_id,
1032        "title": new_title,
1033        "status": "pending",
1034        "created_at": now.isoformat(),
1035    }
