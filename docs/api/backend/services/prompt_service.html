backend.services.prompt_service

Prompt service — variant management with hot-reload.

Wraps prompt file loading with caching based on file modification time. Supports variant overrides and fallback to default prompts.

When a ProfileService is provided, DB content (from blueprints.db) is checked first — this is the Single Source of Truth. Filesystem prompts serve as fallback for project-specific overrides or when the DB is empty.

 56    def get_prompt(
 57        self,
 58        variant: str,
 59        role: str,
 60        language: str = "de",
 61        project_dir: Path | str | None = None,
 62    ) -> dict:
 63        """Load a prompt template with caching and hot-reload.
 64
 65        When a ``ProfileService`` is attached, DB content is checked first
 66        (Single Source of Truth).  Falls back to filesystem for project-
 67        specific overrides or when DB has no matching template.
 68
 69        For non-default languages (e.g. 'en'), tries ``{role}-{lang}.md``
 70        first, then falls back to ``{role}.md``.
 71
 72        Returns a dict with keys: content, hash, mtime, path (mtime for filesystem).
 73        """
 74        # 1. Try DB content via ProfileService (SSOT)
 75        if self._profile_service is not None:
 76            db_data = self._profile_service.get_prompt_content(
 77                variant,
 78                role,
 79                language=language,
 80            )
 81            if db_data:
 82                return db_data
 83
 84        # 2. Fallback: filesystem with hot-reload caching
 85        # Build candidate file names: language-specific first, then base
 86        candidates = []
 87        if language:
 88            candidates.append(f"{role}-{language}.md")
 89        candidates.append(f"{role}.md")
 90
 91        # Try project-specific prompts first
 92        prompt_path = None
 93        if project_dir is not None:
 94            project_prompts = Path(project_dir) / "prompts"
 95            if variant == "default":
 96                project_base = project_prompts / "default"
 97            else:
 98                project_base = project_prompts / "variants" / variant
 99            for name in candidates:
100                path = project_base / name
101                if path.exists():
102                    prompt_path = path
103                    break
104
105        # Determine global base directory
106        if prompt_path is None:
107            if variant == "default":
108                base_dir = self.prompts_dir / "default"
109            else:
110                base_dir = self.prompts_dir / "variants" / variant
111
112            # Try candidates in order
113            for name in candidates:
114                path = base_dir / name
115                if path.exists():
116                    prompt_path = path
117                    break
118
119            # Fallback to default variant if variant-specific not found
120            if prompt_path is None:
121                default_dir = self.prompts_dir / "default"
122                for name in candidates:
123                    path = default_dir / name
124                    if path.exists():
125                        logger.warning(
126                            "Prompt %s/%s not found, falling back to default/%s",
127                            variant,
128                            role,
129                            name,
130                        )
131                        prompt_path = path
132                        break
133
134        if prompt_path is None:
135            raise FileNotFoundError(f"Prompt not found: {variant}/{role} (language={language})")
136
137        cache_key = f"{variant}/{role}/{language}"
138
139        current_mtime = prompt_path.stat().st_mtime
140
141        with self._lock:
142            cached = self._cache.get(cache_key)
143            if cached and cached["mtime"] == current_mtime:
144                return cached
145
146            # Hot-reload
147            content = prompt_path.read_text(encoding="utf-8")
148            data = {
149                "content": content,
150                "hash": hashlib.sha256(content.encode()).hexdigest()[:16],
151                "mtime": current_mtime,
152                "path": str(prompt_path),
153            }
154            self._cache[cache_key] = data
155            logger.info("Prompt loaded: %s/%s (hash=%s)", variant, role, data["hash"])
156            return data

161        role: str,
162        variables: dict[str, str] | None = None,
163        language: str = "de",
164        project_dir: Path | str | None = None,

166        """Load a prompt and optionally substitute variables.

168        Variables are replaced using simple {key} syntax.
169        """
170        data = self.get_prompt(variant, role, language=language, project_dir=project_dir)
171        content = data["content"]
172
173        if variables:
174            for key, value in variables.items():
175                content = content.replace(f"{{{key}}}", value)
176
177        return content

221    def get_prompt_translated(
222        self,
223        variant: str,
224        role: str,
225        target_language: str,
226        project_dir: Path | str | None = None,
227        force_translation: bool = False,
228    ) -> TranslationEntry:
229        """Get a prompt and translate it, returning a TranslationEntry.
230
231        Checks translation cache first. Falls back to source content.
232        """
233        source_data = self.get_prompt(variant, role, language="en", project_dir=project_dir)
234        cache_key = f"{variant}/{role}/{target_language}"
235
236        # Check translation cache first
237        with self._lock:
238            cached = self._cache.get(cache_key)
239            if cached and not force_translation:
240                return cached
241
242        # Use TranslationService for LLM-based translation
243        try:
244            trans_svc = _get_translation_service()
245            entry = trans_svc.get_prompt_translated(
246                module_id="prompts-base",
247                file_path=f"{variant}/{role}",
248                target_language=target_language,
249                source_content=source_data.get("content", ""),
250                source_hash=source_data.get("hash", ""),
251                force=force_translation,
252            )
253
254            if entry and entry.translated_content:
255                result = TranslationEntry(
256                    module_id="prompts-base",
257                    file_path=f"{variant}/{role}",
258                    source_language="en",
259                    target_language=target_language,
260                    translated_content=entry.translated_content,
261                    quality_score=entry.quality_score,
262                    approved=entry.approved,
263                )
264                with self._lock:
265                    self._cache[cache_key] = result
266                return result
267        except Exception:
268            logger.exception(
269                "Translation failed for %s/%s → %s, falling back to English content",
270                variant,
271                role,
272                target_language,
273            )
274
275        # Fallback: return English source content when translation unavailable
276        logger.warning(
277            "Using English fallback for %s/%s (requested language: %s)",
278            variant,
279            role,
280            target_language,
281        )
282        result = TranslationEntry(
283            module_id="prompts-base",
284            file_path=f"{variant}/{role}",
285            source_language="en",
286            target_language=target_language,
287            translated_content=source_data.get("content", ""),
288            quality_score=0.0,
289            approved=False,
290        )
291        return result

293    def translate_prompt(
294        self,
295        variant: str,
296        role: str,
297        target_language: str = "de",
298        force: bool = False,
299        auto_approve: bool = True,
300        quality_threshold: float = 0.7,
301    ) -> dict:
302        """Translate a specific prompt using the TranslationService.
303
304        Args:
305            variant: Prompt variant (e.g. "default")
306            role: Role name (e.g. "strategist")
307            target_language: Target language code
308            force: Force re-translation
309            auto_approve: Auto-approve if quality meets threshold
310            quality_threshold: Quality threshold for auto-approval
311
312        Returns:
313            Dict with translation result details
314        """
315        trans_svc = _get_translation_service()
316        # Import source content first
317        source_data = self.get_prompt(variant, role, language="en")
318        trans_svc.import_source_content(
319            module_id="prompts-base",
320            file_path=f"{variant}/{role}",
321            content=source_data.get("content", ""),
322        )
323        # Perform translation
324        result = trans_svc.translate_module(
325            module_id="prompts-base",
326            target_language=target_language,
327            force=force,
328            auto_approve=auto_approve,
329            quality_threshold=quality_threshold,
330        )
331        return {
332            "module_id": result.module_id,
333            "target_language": result.target_language,
334            "files_translated": result.files_translated,
335            "files_skipped": result.files_skipped,
336            "status": result.status,
337            "quality_scores": result.quality_scores,
338        }

340    def assemble_prompt(
341        self,
342        role_type_id: str,
343        argumentation_pattern: str | None = None,
344        workflow_variant: str = "default",
345        language: str = "de",
346        translate: bool = False,
347    ) -> str:
348        """Assemble the full system prompt from layers:
349
350        1. Argumentation pattern base (if set)
351        2. Workflow-variant prompt overlay
352        3. Fallback to default variant if workflow_variant not found
353
354        Args:
355            role_type_id: The role type (strategist, critic, optimizer, moderator)
356            argumentation_pattern: Optional argumentation pattern name
357            workflow_variant: Workflow variant (default, kantian, steiner, etc.)
358            language: Target language for the prompt (de|en)
359            translate: If True, use TranslationService for translation
360        """
361        parts: list[str] = []
362
363        # Layer 1: Argumentation pattern
364        if argumentation_pattern:
365            if translate and language != "en":
366                ap_prompt = self.get_prompt_translated(
367                    argumentation_pattern,
368                    role_type_id,
369                    target_language=language,
370                ).translated_content
371                if ap_prompt:
372                    parts.append(ap_prompt)
373            else:
374                ap_prompt = self.get_argumentation_pattern(
375                    argumentation_pattern,
376                    role_type_id,
377                    language,
378                )
379                if ap_prompt:
380                    parts.append(ap_prompt)
381
382        # Layer 2: Workflow-variant prompt
383        try:
384            if translate and language != "en":
385                wf_data = self.get_prompt_translated(
386                    workflow_variant,
387                    role_type_id,
388                    target_language=language,
389                )
390                wf_prompt = wf_data.translated_content
391            else:
392                wf_prompt_obj = self.get_prompt(workflow_variant, role_type_id, language=language)
393                wf_prompt = wf_prompt_obj.get("content")
394            if wf_prompt:
395                parts.append(wf_prompt)
396        except FileNotFoundError:
397            try:
398                if translate and language != "en":
399                    wf_data = self.get_prompt_translated(
400                        "default",
401                        role_type_id,
402                        target_language=language,
403                    )
404                    wf_prompt = wf_data.translated_content
405                else:
406                    wf_prompt_obj = self.get_prompt("default", role_type_id, language=language)
407                    wf_prompt = wf_prompt_obj.get("content")
408                if wf_prompt:
409                    parts.append(wf_prompt)
410            except FileNotFoundError as e:
411                logger.warning(
412                    "Prompt template not found for %s/%s (language=%s, translate=%s): %s",
413                    workflow_variant,
414                    role_type_id,
415                    language,
416                    translate,
417                    e,
418                )
419
420        return "\n\n".join(parts)

def get_prompt( self, variant: str, role: str, language: str = 'de', project_dir: pathlib.Path | str | None = None) -> dict: 
 57    def get_prompt(
 58        self,
 59        variant: str,
 60        role: str,
 61        language: str = "de",
 62        project_dir: Path | str | None = None,
 63    ) -> dict:
 64        """Load a prompt template with caching and hot-reload.
 65
 66        When a ``ProfileService`` is attached, DB content is checked first
 67        (Single Source of Truth).  Falls back to filesystem for project-
 68        specific overrides or when DB has no matching template.
 69
 70        For non-default languages (e.g. 'en'), tries ``{role}-{lang}.md``
 71        first, then falls back to ``{role}.md``.
 72
 73        Returns a dict with keys: content, hash, mtime, path (mtime for filesystem).
 74        """
 75        # 1. Try DB content via ProfileService (SSOT)
 76        if self._profile_service is not None:
 77            db_data = self._profile_service.get_prompt_content(
 78                variant,
 79                role,
 80                language=language,
 81            )
 82            if db_data:
 83                return db_data
 84
 85        # 2. Fallback: filesystem with hot-reload caching
 86        # Build candidate file names: language-specific first, then base
 87        candidates = []
 88        if language:
 89            candidates.append(f"{role}-{language}.md")
 90        candidates.append(f"{role}.md")
 91
 92        # Try project-specific prompts first
 93        prompt_path = None
 94        if project_dir is not None:
 95            project_prompts = Path(project_dir) / "prompts"
 96            if variant == "default":
 97                project_base = project_prompts / "default"
 98            else:
 99                project_base = project_prompts / "variants" / variant
100            for name in candidates:
101                path = project_base / name
102                if path.exists():
103                    prompt_path = path
104                    break
105
106        # Determine global base directory
107        if prompt_path is None:
108            if variant == "default":
109                base_dir = self.prompts_dir / "default"
110            else:
111                base_dir = self.prompts_dir / "variants" / variant
112
113            # Try candidates in order
114            for name in candidates:
115                path = base_dir / name
116                if path.exists():
117                    prompt_path = path
118                    break
119
120            # Fallback to default variant if variant-specific not found
121            if prompt_path is None:
122                default_dir = self.prompts_dir / "default"
123                for name in candidates:
124                    path = default_dir / name
125                    if path.exists():
126                        logger.warning(
127                            "Prompt %s/%s not found, falling back to default/%s",
128                            variant,
129                            role,
130                            name,
131                        )
132                        prompt_path = path
133                        break
134
135        if prompt_path is None:
136            raise FileNotFoundError(f"Prompt not found: {variant}/{role} (language={language})")
137
138        cache_key = f"{variant}/{role}/{language}"
139
140        current_mtime = prompt_path.stat().st_mtime
141
142        with self._lock:
143            cached = self._cache.get(cache_key)
144            if cached and cached["mtime"] == current_mtime:
145                return cached
146
147            # Hot-reload
148            content = prompt_path.read_text(encoding="utf-8")
149            data = {
150                "content": content,
151                "hash": hashlib.sha256(content.encode()).hexdigest()[:16],
152                "mtime": current_mtime,
153                "path": str(prompt_path),
154            }
155            self._cache[cache_key] = data
156            logger.info("Prompt loaded: %s/%s (hash=%s)", variant, role, data["hash"])
157            return data
