220def _post_process_title(raw: str, case_text: str) -> str:
221    """Clean and validate a generated title, falling back if needed."""
222    _re = __import__("re")
223    verbose = [
224        r"^Here(?:'s| is) (?:a |the |an )?(?:suggested |proposed )?(?:debate )?title[:\s]*",
225        r"^The title is[:\s]*",
226        r"^Title[:\s]*",
227        r"^I suggest[:\s]*",
228        r"^I propose[:\s]*",
229        r"^I think[:\s]*",
230        r"^Der Titel lautet[:\s]*",
231        r"^Titel[:\s]*",
232        r"^Hier ist ein Titel[:\s]*",
233        r"^Ich schlage vor[:\s]*",
234        r"^Ein passender Titel (?:wäre|ist)[:\s]*",
235    ]
236    cleaned = raw.strip()
237    for pat in verbose:
238        cleaned = _re.sub(pat, "", cleaned, count=1, flags=_re.IGNORECASE).strip()
239    for ch in ['"', "'", "„", """, """, """, """, "`", "-", "–", ":", ".", ","]:
240        cleaned = cleaned.strip(ch)
241    cleaned = cleaned.strip()
242
243    if not cleaned or len(cleaned) < 15 or len(cleaned) > 150:
244        return _fallback_title(case_text)
245
246    reflection = [
247        r"\b(?:the user|der benutzer)\b",
248        r"\bbased on\b",
249        r"\bbasierend auf\b",
250        r"\b(?:description|beschreibung)\b.*\b(?:about|um)\b",
251        r"^-\s",
252    ]
253    if any(_re.search(p, cleaned, _re.IGNORECASE) for p in reflection):
254        return _fallback_title(case_text)
255
256    # Final validation using standalone validate_title()
257    valid, reason = validate_title(cleaned, case_text)
258    if not valid:
259        logger.debug("Title validation failed: %s — %s", reason, cleaned[:80])
260        return _fallback_title(case_text)
261
262    return cleaned

 92def _select_service_llm(profile_service) -> str:
 93    """Select the service LLM profile for system/background tasks.
 94
 95    Strategy — respects the user's UI choice (Build > Manage > LLM Profile,
 96    Utility-LLM checkbox) above all else:
 97    1. User-saved profiles (not readonly) with ``service_eligible`` = True.
 98    2. ``settings.service_llm_profile_id`` if it matches an eligible user-saved profile.
 99    3. Any eligible readonly profile.
100    4. First available text-LLM.
101    """
102    try:
103        all_profiles = profile_service.list_llm_profiles()
104    except Exception:
105        raise RuntimeError("No suitable LLM profile available for service tasks")
106
107    # --- User-saved eligible profiles (UI checkbox choice) ---
108    user_saved = [p for p in all_profiles if is_service_llm_eligible(p)[0] and not getattr(p, "_readonly", False)]
109    if user_saved:
110        if settings.service_llm_profile_id:
111            match = next((p for p in user_saved if p.id == settings.service_llm_profile_id), None)
112            if match:
113                return match.id
114        user_saved.sort(key=lambda p_: -(p_.context_window or 0))
115        return user_saved[0].id
116
117    # --- settings.service_llm_profile_id (fallback) ---
118    if settings.service_llm_profile_id:
119        preferred = profile_service.get_llm_profile(settings.service_llm_profile_id)
120        if preferred and is_service_llm_eligible(preferred)[0]:
121            return settings.service_llm_profile_id
122
123    # --- Any eligible readonly profile ---
124    readonly_eligible = [p for p in all_profiles if is_service_llm_eligible(p)[0]]
125    if readonly_eligible:
126        readonly_eligible.sort(key=lambda p_: -(p_.context_window or 0))
127        return readonly_eligible[0].id
128
129    # --- Last resort ---
130    text_profiles = [p for p in all_profiles if getattr(p, "profile_type", "text") == "text"]
131    if text_profiles:
132        return text_profiles[0].id
133
134    raise RuntimeError("No suitable LLM profile available for service tasks")
